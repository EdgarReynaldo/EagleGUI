204 void PinLayout::CyclePins(bool cycle_forward) {
205  if (pins.size()) {
206  if (cycle_forward) {
207  Pin ptemp = pins.back();
208  for (int i = (int)pins.size() - 2 ; i >= 0 ; --i) {
209  pins[i + 1] = pins[i];
