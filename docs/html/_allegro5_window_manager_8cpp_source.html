23 void* A5WindowManagerProcess(EagleThread* thread , void* manager) {
24 
25  Allegro5WindowManager* a5man = dynamic_cast<Allegro5WindowManager*>((EagleWindowManager*)manager);
26  EAGLE_ASSERT(a5man);
27 
28  ALLEGRO_EVENT_QUEUE* window_queue = a5man->window_queue;
29 
30  while (1) {
31  if (thread->ShouldStop()) {
32  EagleInfo() << "A5WindowManagerProcess - thread has been signaled to stop." << std::endl;
33  break;
34  }
35 
36  EAGLE_ASSERT(al_is_event_source_registered(window_queue , &(a5man->window_event_source)));
37 
38  ALLEGRO_EVENT ev;
39  al_wait_for_event(window_queue , &ev);
40 
41  EagleEvent ee = a5man->GetEagleDisplayEvent(ev);/** This LOCKS the wm mutex **/
42  EagleInfo() << "Allegro5WindowManagerProcess received event " << EagleEventName(ee.type) << std::endl;
43 
44  if (ev.any.source == &(a5man->window_event_source)) {
45  /** Special message , either EAGLE_EVENT_WM_CLOSE_WINDOW, or EAGLE_EVENT_WM_DESTROY */
46  a5man->EmitEvent(ee , thread);
47  if (ev.type == EAGLE_EVENT_WM_DESTROY) {
48  EAGLE_ASSERT(a5man->WindowCount() == 0);
49  EagleInfo() << "EAGLE_EVENT_WM_DESTROY received by A5WindowManagerProcess" << std::endl;
50  break;
51  }
52  else if (ev.type == EAGLE_EVENT_WM_CLOSE_WINDOW) {
53 
54  Allegro5GraphicsContext* win = (Allegro5GraphicsContext*)ev.user.data1;
55  EAGLE_ASSERT(win);
56  ALLEGRO_DISPLAY* d = win->AllegroDisplay();
57  EAGLE_ASSERT(d);
58 
59  EagleInfo() << "EAGLE_EVENT_WM_CLOSE_WINDOW received by A5WindowManagerProcess" << std::endl;
60  EagleInfo() << StringPrintF("A5WindowManagerProcess : Closing display %p" , d) << std::endl;
61 
62  al_unregister_event_source(window_queue , al_get_display_event_source(d));
63 
64  ALLEGRO_EVENT_SOURCE* response_event_source = &(a5man->response_event_source);
65 
66  ALLEGRO_EVENT ev2;
67  ev2.type = EAGLE_EVENT_WM_CLOSE_WINDOW_RECEIVED;
68  ev2.user.data1 = (intptr_t)win;
69  al_emit_user_event(response_event_source , &ev2 , 0);
70  }
71  continue;
72  }
73 
74  if (ee.type == EAGLE_EVENT_DISPLAY_SWITCH_IN) {
75  a5man->SwitchIn(ee.window);
76  }
77  else if (ee.type == EAGLE_EVENT_DISPLAY_SWITCH_OUT) {
78  a5man->SwitchOut(ee.window);
79  }
80 
81  EAGLE_ASSERT(ee.window);
82  /// This should always be a window event
83  ee.window->EmitEvent(ee , thread);
84  }
85  return 0;
86 }

152 EagleEvent Allegro5WindowManager::GetEagleDisplayEvent(ALLEGRO_EVENT ev) {
153  EagleEvent ee;
154 
155  Allegro5GraphicsContext* a5win = 0;
156 
157  if (ev.any.source == &window_event_source) {
158  /// We sent this message to ourselves
159  /// Event should be EAGLE_EVENT_WM_CLOSE_WINDOW or EAGLE_EVENT_WM_DESTROY
160  if (ev.type == EAGLE_EVENT_WM_CLOSE_WINDOW) {
161  a5win = (Allegro5GraphicsContext*)ev.user.data1;
162  ee.window = a5win;
163  ee.type = (EAGLE_EVENT_TYPE)EAGLE_EVENT_WM_CLOSE_WINDOW;
164  ee.display.x = a5win->XPos();
165  ee.display.y = a5win->YPos();
166  ee.display.width = a5win->Width();
167  ee.display.height = a5win->Height();
168  ee.display.orientation = al_get_display_orientation(a5win->AllegroDisplay());
169  return ee;
170  }
171  else if (ev.type == EAGLE_EVENT_WM_DESTROY) {
172 /// Allegro5WindowManager wm = (Allegro5WindowManager*)ev.user.data1;
173  ee.type = (EAGLE_EVENT_TYPE)EAGLE_EVENT_WM_DESTROY;
174  ee.window = 0;
175  return ee;
176  }
177  else {
178  throw EagleException("Unknown event from window event source in Allegro5WindowManager::GetEagleDisplayEvent()");
179  }
180  }
181  else {
182  /** THIS HAS TO BE A DISPLAY EVENT!**/
183  a5win = dynamic_cast<Allegro5GraphicsContext*>(GetAssociatedContext(ev.display.source));/** This LOCKS the wm mutex */
184 
185  ee.display.x = ev.display.x;
186  ee.display.y = ev.display.y;
187  ee.display.width = ev.display.width;
188  ee.display.height = ev.display.height;
189  ee.display.orientation = ev.display.orientation;
190  }
191 
192  EAGLE_ASSERT(a5win);
193 
194  ee.source = a5win;
195  ee.window = a5win;
196 
197  switch(ev.type) {
198  case ALLEGRO_EVENT_DISPLAY_EXPOSE :
199  ee.type = EAGLE_EVENT_DISPLAY_EXPOSE;
200  break;
201  case ALLEGRO_EVENT_DISPLAY_RESIZE :
202  ee.type = EAGLE_EVENT_DISPLAY_RESIZE;
203  break;
204  case ALLEGRO_EVENT_DISPLAY_CLOSE :
205  ee.type = EAGLE_EVENT_DISPLAY_CLOSE;
206  break;
207  case ALLEGRO_EVENT_DISPLAY_LOST :
208  ee.type = EAGLE_EVENT_DISPLAY_LOST;
209  break;
210  case ALLEGRO_EVENT_DISPLAY_FOUND :
211  ee.type = EAGLE_EVENT_DISPLAY_FOUND;
212  break;
213  case ALLEGRO_EVENT_DISPLAY_SWITCH_OUT :
214  ee.type = EAGLE_EVENT_DISPLAY_SWITCH_OUT;
215  break;
216  case ALLEGRO_EVENT_DISPLAY_SWITCH_IN :
217  ee.type = EAGLE_EVENT_DISPLAY_SWITCH_IN;
218  break;
219  case ALLEGRO_EVENT_DISPLAY_ORIENTATION :
220  ee.type = EAGLE_EVENT_DISPLAY_ORIENTATION;
221  break;
222  case ALLEGRO_EVENT_DISPLAY_HALT_DRAWING :
223  ee.type = EAGLE_EVENT_DISPLAY_HALT_DRAWING;
224  break;
225  case ALLEGRO_EVENT_DISPLAY_RESUME_DRAWING :
226  ee.type = EAGLE_EVENT_DISPLAY_RESUME_DRAWING;
227  break;
228 
229  default :
230  throw EagleException(StringPrintF("A5WindowProcess : We don't monitor this kind of event (%d)." , ev.type));
231  break;
232  }
233  return ee;
234 }

268 bool Allegro5WindowManager::Create() {
269 
270  Destroy();
271 
272  al_init_user_event_source(&window_event_source);
273  al_init_user_event_source(&response_event_source);
274 
275  window_queue = al_create_event_queue();
276  response_queue = al_create_event_queue();
277 
278  if (!window_queue) {
279  throw EagleException("Allegro5WindowManager::Create - Failed to create allegro event queue.");
280  }
281  if (!response_queue) {
282  throw EagleException("Allegro5WindowManager::Create - Failed to create allegro event queue.");
283  }
284 
285  al_register_event_source(window_queue , &window_event_source);
286  al_register_event_source(response_queue , &response_event_source);
287 
288  manager_mutex = new CXX11Mutex("A5WM::manager_mutex");
289 
290  manager_thread = new Allegro5Thread("A5WM::manager_thread");
291 
292  if (!manager_mutex->Create(true , false)) {
293  throw EagleException("Allegro5WindowManager::Allegro5WindowManager - failed to create the window manager mutex.");
294  }
295 
296  if (!manager_thread->Create(A5WindowManagerProcess , this)) {
297  throw EagleException("Allegro5WindowManager::Allegro5WindowManager - failed to create the window manager thread.");
298  }
299 
300  manager_thread->Start();
301 
302  return window_queue && manager_mutex && manager_mutex->Valid() && manager_thread && manager_thread->Running();
303 }
