66 void RadioGroup::RespondToEvent(EagleEvent e , EagleThread* thread) {
67  if (e.type == EAGLE_EVENT_WIDGET) {
68  if (e.widget.topic == TOPIC_BUTTON_WIDGET && e.widget.msgs == BUTTON_TOGGLED) {
69  BasicButton* btn = dynamic_cast<BasicButton*>(e.widget.from);
70  if (btn) {
71  if (btn_group.find(btn) != btn_group.end()) {
72  /// The button is in our set, and it was toggled
73  bool up = btn->Up();
74  if (!up) {
75  SelectButton(btn);
76  e.source = this;
77  EmitEvent(e , thread);
78  }
79  }
80  }
