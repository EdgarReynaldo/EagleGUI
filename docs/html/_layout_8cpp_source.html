137 void LayoutBase::AdjustWidgetArea(const WidgetBase* widget , int* newx , int* newy , int* newwidth , int* newheight) const {
138  EAGLE_ASSERT(widget);
139  EAGLE_ASSERT(newx);
140  EAGLE_ASSERT(newy);
141  EAGLE_ASSERT(newwidth);
142  EAGLE_ASSERT(newheight);
143 
144  unsigned int flags = (unsigned int)widget->Flags();
145 
146  Rectangle w = widget->OuterArea();
147 
148  if (*newx == INT_MAX) {
149  *newx = w.X();
150  }
151  if (*newy == INT_MAX) {
152  *newy = w.Y();
153  }
154  if (*newwidth == INT_MAX) {
155  *newwidth = w.W();
156  }
157  else {
158  int mw = widget->AbsMinWidth();
159  if (*newwidth < abs(mw)) {
160  *newwidth = mw;
161  }
162  }
163  if (*newheight == INT_MAX) {
164  *newheight = w.H();
165  }
166  else {
167  int mh = widget->AbsMinHeight();
168  if (*newheight < abs(mh)) {
169  *newheight = mh;
170  }
171  }
172 
173  if (!(flags & MOVEABLE)) {
174  *newx = w.X();
175  *newy = w.Y();
176  }
177  if (!(flags & RESIZEABLE)) {
178  *newwidth = w.W();
179  *newheight = w.H();
180  }
181 
182  if (!(attributes & LAYOUT_ALLOWS_REPOSITION)) {
183  *newx = w.X();
184  *newy = w.Y();
185  }
186  if (!(attributes & LAYOUT_ALLOWS_RESIZE)) {
187  *newwidth = w.W();
188  *newheight = w.H();
189  }
190 
191  /// Handle rectangles with negative width or height
192  Rectangle r;
193  r.SetCorners(*newx , *newy , *newx + *newwidth - 1 , *newy + *newheight - 1);
194  *newx = r.X();
195  *newy = r.Y();
196  *newwidth = (r.W()>0)?r.W():1;
197  *newheight = (r.H()>0)?r.H():1;
198 
199 }
