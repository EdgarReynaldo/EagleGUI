247 void Triangle::RenewAngles() {
248  CheckPoints();
249  if (points_unique) {
250  angle_ab = p1.AngleToPoint(p2);// -PI to PI
251  angle_bc = p2.AngleToPoint(p3);// -PI to PI
252  angle_ca = p3.AngleToPoint(p1);// -PI to PI
253  double turn_angle = angle_bc - angle_ab;//[-PI - PI , PI - -PI] = [-2PI , 2PI]
254  // Normalize turn angle to (-M_PI , M_PI];
255  if (turn_angle <= -M_PI) {turn_angle += 2.0*M_PI;}
256  if (turn_angle > M_PI) {turn_angle -= 2.0*M_PI;};
257  if ((turn_angle == 0.0) || (turn_angle == M_PI)) {
258  // collinear points
259  points_make_triangle = false;
260  }
261  inside_is_positive_angle = (turn_angle > 0.0);
262  }

313 bool Triangle::Contains(int xpos , int ypos) const {
314  if (!points_make_triangle) {return false;}
315  Pos2D p4(xpos,ypos);
316  const Pos2D* pts[3] = {&p1 , &p2 , &p3};
317  double angles[3] = {angle_ab , angle_bc , angle_ca};
318  double turn_angle = 0.0;
319  double angle_to_p4 = 0.0;
320  for (int i = 0 ; i < 3 ; ++i) {
321  if (*pts[i] == p4) {return true;}
322  angle_to_p4 = (*pts[i]).AngleToPoint(p4);// -PI to PI
323  turn_angle = angle_to_p4 - angles[i];// -2PI to 2PI
324  // Normalize turn angle to (-M_PI , M_PI];
325  if (turn_angle <= -M_PI) {turn_angle += 2.0*M_PI;}
326  if (turn_angle > M_PI) {turn_angle -= 2.0*M_PI;};
327  if (inside_is_positive_angle && (turn_angle < 0.0)) {return false;}
328  if (!inside_is_positive_angle && (turn_angle > 0.0)) {return false;}
329  }
330  return true;
331 }

1173 list<Rectangle> CombineRectangles(Rectangle r1 , Rectangle r2) {
1174  list<Rectangle> rlist;
1175 
1176  // Contained Rectangles
1177  RECT_CONTAINS_TYPE contype = DoRectanglesContain(r1,r2);
1178  if (contype) {
1179  if (contype == R1_CONTAINS_R2) {rlist.push_back(r1);}
1180  if (contype == R2_CONTAINS_R1) {rlist.push_back(r2);}
1181  return rlist;
1182  }
1183 
1184  // Connecting Rectangles
1185  RECT_CONNECT_TYPE rctype = DoRectanglesConnect(r1,r2);
1186  if (rctype) {
1187  if (rctype == RECT_CONNECT_R2_ON_TOP) {
1188  rlist.push_back(Rectangle(r2.X() , r2.Y() , r2.W() , r1.H() + r2.H()));
1189  }
1190  if (rctype == RECT_CONNECT_R2_ON_BOTTOM) {
1191  rlist.push_back(Rectangle(r1.X() , r1.Y() , r1.W() , r1.H() + r2.H()));
1192  }
1193  if (rctype == RECT_CONNECT_R2_ON_RIGHT) {
1194  rlist.push_back(Rectangle(r1.X() , r1.Y() , r1.W() + r2.W() , r1.H()));
1195  }
1196  if (rctype == RECT_CONNECT_R2_ON_LEFT) {
1197  rlist.push_back(Rectangle(r2.X() , r2.Y() , r1.W() + r2.W() , r2.H()));
1198  }
1199  return rlist;
1200  }
1201 
1202  // Overlapping Rectangles
1203  /// w = brx - x + 1
1204  /// h = bry - y + 1
1205  if (r1.Overlaps(r2)) {
1206  // Overlapping aligned Rectangles combine to produce one Rectangle
1207  // Vertical alignment
1208  if ((r1.X() == r2.X()) && (r1.BRX() == r2.BRX())) {
1209  const Rectangle* top = (r1.Y() < r2.Y())?&r1:&r2;
1210  const Rectangle* bot = (r1.BRY() > r2.BRY())?&r1:&r2;
1211  rlist.push_back(Rectangle(top->X() , top->Y() , top->W() , bot->BRY() - top->Y() + 1));
1212  return rlist;
1213  }
1214  // Horizontal alignment
1215  if ((r1.Y() == r2.Y()) && (r1.BRY() == r2.BRY())) {
1216  const Rectangle* left = (r1.X() < r2.X())?&r1:&r2;
1217  const Rectangle* right = (r1.BRX() > r2.BRX())?&r1:&r2;
1218  rlist.push_back(Rectangle(left->X() , left->Y() , right->BRX() - left->X() + 1 , left->H()));
1219  return rlist;
1220  }
1221  // Overlapping offset Rectangles
1222  const Rectangle* temin = (r1.Y() < r2.Y())?&r1:&r2;// top edge min y value
1223  const Rectangle* temax = (r1.Y() > r2.Y())?&r1:&r2;// top edge max y value
1224  const Rectangle* lemin = (r1.X() < r2.X())?&r1:&r2;// left edge min x value
1225  const Rectangle* lemax = (r1.X() > r2.X())?&r1:&r2;// left edge max x value
1226  const Rectangle* bemin = (r1.BRY() < r2.BRY())?&r1:&r2;// bottom edge min y value
1227  const Rectangle* bemax = (r1.BRY() > r2.BRY())?&r1:&r2;// bottom edge max y value
1228  const Rectangle* remin = (r1.BRX() < r2.BRX())?&r1:&r2;// right edge min x value
1229  const Rectangle* remax = (r1.BRX() > r2.BRX())?&r1:&r2;// right edge max x value
1230  if (r1.Y() == r2.Y()) {
1231  // When the tops are even, only two unique Rectangles are formed
1232  // new top Rectangle formed by min left edge , max right edge , either top edge , and the min bottom edge
1233  rlist.push_back(Rectangle(lemin->X() , r1.Y() , remax->BRX() - lemin->X() + 1 , bemin->BRY() - r1.Y() + 1));
1234  // new bot Rectangle formed by bottom Rectangles left and right edges , top.BRY() + 1 , and bot.BRY()
1235  rlist.push_back(Rectangle(bemax->X() , bemin->BRY() + 1 , bemax->BRX() - bemax->X() + 1 , bemax->BRY() - (bemin->BRY() + 1) + 1));
1236  } else if (r1.BRY() == r2.BRY()) {
1237  // When the bottoms are even, only two unique Rectangles are formed
1238  // new top Rectangle formed by top Rectangles left and right edges , top.Y() and bot.Y() - 1
1239  rlist.push_back(Rectangle(temin->X() , temin->Y() , temin->BRX() - temin->X() + 1 , (temax->Y() - 1) - temin->Y() + 1));
1240  // new bot Rectangle formed by min left edge , max right edge , bot.Y() and either bottom edge
1241  rlist.push_back(Rectangle(lemin->X() , temax->Y() , remax->BRX() - lemin->X() + 1 , r1.BRY() - temax->Y() + 1));
1242  } else if (r1.X() == r2.X()) {
1243  // When the left edges are even, two vertical Rectangles are formed
