18  return ret;
19  }

22  if (e.keyboard.keycode == EAGLE_KEY_INSERT) {
23  /// Toggle insert mode

25  SetRedrawFlag();

27  }
28  }
29  else if (e.type == EAGLE_EVENT_KEY_CHAR) {

32  if (e.keyboard.unicode <= 27 || e.keyboard.unicode == 127) {
33  /// Control key, or invisible
34  if (e.keyboard.keycode == EAGLE_KEY_TAB) {
35  newstr = "";
36  for (unsigned int i = 0 ; i < NTABSPACES ; ++i) {
37  newstr += " ";
38  }
39  /// TODO : Follow indentation guides
40  }
41  else if (e.keyboard.keycode == EAGLE_KEY_ENTER) {
42  newstr = '\n';
43  }
44  else if (e.keyboard.keycode == EAGLE_KEY_DELETE) {
45  if (select_line_start > -1) {
46  DeleteSelection(select_line_start , select_line_close , select_left , select_right);
47  }
48  else {
49  if (caret_pos < (int)lines[caret_line].size()) {
50  DeleteSelection(caret_line , caret_line , caret_pos , caret_pos + 1);
51  }
52  }
53  Refresh();
54  ResetSelection();
55  return DIALOG_INPUT_USED;
56  }
57  else if (e.keyboard.keycode == EAGLE_KEY_BACKSPACE) {
58  if (select_line_start > -1) {
59  DeleteSelection(select_line_start , select_line_close , select_left , select_right);
60  }
61  else {
62  if (caret_pos > 0) {
63  DeleteSelection(caret_line , caret_line , caret_pos - 1 , caret_pos);
64 
65  }
66  else if (caret_pos == 0) {
67  if (caret_line > 0) {
68  DeleteSelection(caret_line - 1 , caret_line , lines[caret_line-1].size() - 1 , 0);
69  }
70  }
71  }
72  Refresh();
73  ResetSelection();
74  return DIALOG_INPUT_USED;
75  }
76  else if (e.keyboard.keycode == EAGLE_KEY_V && input_key_held(EAGLE_KEY_ANY_CTRL)) {
77  /// Paste action
78 
79  /// TODO : This won't work for stand alone widgets - it depends on the root gui's drawing window
80  EagleGraphicsContext* win = GetDrawWindow();
81  EAGLE_ASSERT(win);
82 
83  EagleSystem* sys = win->GetSystem();
84  EAGLE_ASSERT(sys);
85 
86  EagleClipboard* cb = sys->GetSystemClipboard();
87  EAGLE_ASSERT(cb);
88 
89  newstr = cb->GetClipboardString();
90  }
91  }
92  else if (e.keyboard.unicode < 127) {
93  newstr.push_back((char)e.keyboard.unicode);
94  }
95 
96  if (newstr.size()) {
97  if (Overwrite()) {
98  /// Overwrite
99  if (select_line_start > -1) {
100  Overwrite(newstr , select_line_start , select_line_close , select_left , select_right);
101  }
102  else {
103  if (caret_pos < (int)lines[caret_line].size()) {
104  Overwrite(newstr , caret_line , caret_line , caret_pos , caret_pos + 1);
105  } else {
106  if (caret_line == (int)lines.size() - 1) {
107  Insert(newstr , caret_line , caret_pos);
108  }
109  }
110  }
111  }
112  else {
113  /// Insert mode
114  Insert(newstr , caret_line , caret_pos);
115  }
116  ResetSelection();
117  }
118  return DIALOG_INPUT_USED;
119  }
120  return DIALOG_OKAY;
121 }
