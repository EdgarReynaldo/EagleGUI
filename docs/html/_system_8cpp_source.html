107 std::string PrintEagleInitState(int state) {
108  std::stringstream ss;
109  if (state) {
110  for (int bitshift = 0 ; bitshift < 11 ; ++bitshift) {
111  int flag = 1 << bitshift;
112  ss << ((state & flag)?"+":"-");
113  ss << eagle_init_state_strs[bitshift + 1];
114  state &= ~flag;
115  if (state) {
116  ss << " | ";
117  }
118  }
119  return ss.str();
120  }
121  return eagle_init_state_strs[0];

126 std::string PrintFailedEagleInitStates(int desired_state , int actual_state) {
127  std::stringstream ss;
128  if (desired_state) {
129 
130  if (actual_state & ~desired_state) {
131  ss << "EXTRA STATES TURNED ON!";
132  }
133 
134  if (desired_state & ~actual_state) {
135  ss << "FAILED STATES = ";
136  }
137  else {
138  return "ALL DESIRED STATES SUCCEEDED TO INITIALIZE";
139  }
140 
141  for (int bitshift = 0 ; bitshift < 11 ; ++bitshift) {
142  int flag = 1 << bitshift;
143 
144  if ((desired_state & flag) && !(actual_state & flag)) {
145  ss << "-" << eagle_init_state_strs[bitshift + 1] << "-";
146  }
147  desired_state &= ~flag;
148  actual_state &= ~flag;
149 
150  if (desired_state ^ actual_state) {
151  ss << " | ";
152  }
153  }
154  return ss.str();
155  }
156  return eagle_init_state_strs[0];
157 }

235 int EagleSystem::Initialize(int state) {
236  int desired_state = state;
237 
238  state |= EAGLE_SYSTEM;/// System is non-optional
239 
240  if (state & EAGLE_SYSTEM) {InitializeSystem();}
241  if (state & EAGLE_IMAGES) {InitializeImages();}
242  if (state & EAGLE_FONTS) {InitializeFonts();}
243  if (state & EAGLE_TTF_FONTS) {InitializeTTFFonts();}
244  if (state & EAGLE_AUDIO) {InitializeAudio();}
245  if (state & EAGLE_SHADERS) {InitializeShaders();}
246  if (state & EAGLE_PRIMITIVES) {InitializePrimitives();}
247 
248  if (state & EAGLE_KEYBOARD) {InstallKeyboard();}
249  if (state & EAGLE_MOUSE) {InstallMouse();}
250  if (state & EAGLE_JOYSTICK) {InstallJoystick();}
251  if (state & EAGLE_TOUCH) {InstallTouch();}
252 
253  FinalizeSystem();
254 
255  PrintFailedEagleInitStates(desired_state , EagleInitState());
256 
257  return EagleInitState();
258 }
