94 NinePatch MakeNinePatch(EagleGraphicsContext* win , EagleImage* src_img , WidgetArea src_area) {
95 
96  EAGLE_ASSERT(win);
97  EAGLE_ASSERT(win->Valid());
98  EAGLE_ASSERT(src_img);
99  EAGLE_ASSERT(src_img->Valid());
100 
101  Rectangle src_outer = src_area.OuterArea();
102  Rectangle src_inner = src_area.InnerArea();
103 
104  Rectangle img_rect = Rectangle(0 , 0 , src_img->W() , src_img->H());
105 
106  if (!img_rect.Contains(src_outer) || !img_rect.Contains(src_inner)) {
107  throw EagleException("MakeNinePatch() - Source image area does not fully contain the nine patch area!");
108  }
109 
110  /// Make sure inner area of source is smaller than outer area
111  EAGLE_ASSERT((src_inner.W() > 0) && (src_inner.W() < src_outer.W()));
112  EAGLE_ASSERT((src_inner.H() > 0) && (src_inner.H() < src_outer.H()));
113 
114  /// Make sure the left and right and top and bottom margins are positive
115  EAGLE_ASSERT((src_area.LeftIndent() > 0) && (src_area.RightIndent() > 0));
116  EAGLE_ASSERT((src_area.TopIndent() > 0) && (src_area.BottomIndent() > 0));
117 
118 /// Don't uncomment following line : Users may want a different area, in case of sprite sheets
119 /// src_area.MoveBy(-(src_area.OuterArea().X()) , -(src_area.OuterArea().Y()));
120 
121 /// WidgetArea img_area;
122 
123 /// img_area.SetOuterArea(0 , 0 , src_img->W() , src_img->H());
124 
125 
126  /// Store relative positions in nplayout
127  LayoutRectangle nplayout(src_outer , src_inner);
128 
129  /// Set the matching relative area on the image
130 /// img_area.SetFractionalInnerArea(nplayout.fx , nplayout.fy , nplayout.fw , nplayout.fh);
131 
132  EagleLog() << "NinePatch source area : " << src_area << std::endl;
133 /// EagleLog() << "IMGAREA :" << std::endl;
134 /// EagleLog() << img_area << std::endl;
135 
136  win->SetCopyBlender();
137  NinePatch np;
138  for (int y = 0 ; y < 3 ; ++y) {
139  for (int x = 0 ; x < 3 ; ++x) {
140  np[y][x] = 0;
141  VCELL_AREA vcell = (VCELL_AREA)y;
142  HCELL_AREA hcell = (HCELL_AREA)x;
143  NPAREA nparea = src_area.BorderNP();
144  Rectangle src = nparea.GetNPCell(hcell , vcell);
145  EagleLog() << StringPrintF("Drawing Image : [%d][%d]" , y , x) << std::endl;
146  EagleLog() << "Src = [" << src << "]" << std::endl;
147  EagleImage* img = win->CreateImage(src.W() , src.H() , VIDEO_IMAGE);
148  EAGLE_ASSERT(img);
149  EAGLE_ASSERT(img->Valid());
150  win->PushDrawingTarget(img);
151  win->DrawTintedRegion(src_img , src , 0.0 , 0.0);
152  win->PopDrawingTarget();
153  np[y][x] = img;
154  }
155  }
156  win->RestoreLastBlendingState();
157 
158  return np;
159 }
