29 int Slider::PrivateHandleEvent(EagleEvent ee) {
30  int ret = DIALOG_OKAY;
31 
32 // EagleLog() << "Event " << EagleEventName(ee.type) << " received.\n";
33 
34 
35  if (ee.type == EAGLE_EVENT_MOUSE_AXES) {
36  if (!drag) {
37  mxstart = ee.mouse.x;
38  mystart = ee.mouse.y;
39  }
40  else {
41  if (horizontal) {
42  SetPercent((double)(slider_start + (invert?-1:1)*(ee.mouse.x - mxstart))/slider_max);
43  }
44  else {
45  SetPercent((double)(slider_start + (invert?-1:1)*(ee.mouse.y - mystart))/slider_max);
46  }
47  return DIALOG_INPUT_USED;
48  }
49  }
50  if (ee.type == EAGLE_EVENT_MOUSE_BUTTON_DOWN || ee.type == EAGLE_EVENT_MOUSE_BUTTON_UP) {
51  if (ee.mouse.button == 1) {
52  ret = handle->HandleEvent(ee);
53  }
54  }
55  if (ee.type == EAGLE_EVENT_WIDGET) {
56  if (ee.widget.from == handle) {
57  if (ee.widget.msgs == BUTTON_CLICKED) {
58  drag = true;
59  slider_start = slider_pos;
60  }
61  else if (ee.widget.msgs == BUTTON_RELEASED) {
62  drag = false;
63  }
64  return DIALOG_OKAY;
65  }
66  }
67  return ret;
68 }

105 void Slider::ResizeButton() {
106  Rectangle inner = warea.InnerArea();
107  if (horizontal) {
108  slider_max = InnerArea().W() - handle->InnerArea().W();
109  if (invert) {
110  handle->SetWidgetArea(WidgetArea(Rectangle(inner.X() + slider_max - slider_pos , inner.Y() , inner.W()/10 , inner.H()) , false));
111  }
112  else {
113  handle->SetWidgetArea(WidgetArea(Rectangle(inner.X() + slider_pos , inner.Y() , inner.W()/10 , inner.H()) , false));
114  }
115  }
116  else {
117  slider_max = InnerArea().H() - handle->InnerArea().H();
118  if (invert) {
119  handle->SetWidgetArea(WidgetArea(Rectangle(inner.X() , inner.Y() + slider_max - slider_pos , inner.W() , inner.H()/10)) , false);
120  }
121  else {
122  handle->SetWidgetArea(WidgetArea(Rectangle(inner.X() , inner.Y() + slider_pos , inner.W() , inner.H()/10)) , false);

171 void Slider::SetPercent(double pct , bool sendmessage) {
172  double old = slider_percent;
173  if (pct < 0.0) {pct = 0.0;}
174  if (pct > 1.0) {pct = 1.0;}
175  slider_percent = pct;
176  slider_pos = GetSliderValue(pct);
177  ResizeButton();
178  SetRedrawFlag();
179  if (sendmessage && (old != slider_percent)) {
