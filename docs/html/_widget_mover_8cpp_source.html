30 int WidgetMover::PrivateHandleEvent(EagleEvent e) {
31  if (false) {///!active) {
32  if (hotkey) {
33  active = true;
34  return DIALOG_INPUT_USED;
35  }
36  return DIALOG_OKAY;
37  }
38  if (!IsMouseEvent(e)) {return DIALOG_OKAY;}
39 
40  const Pos2I mabs = AbsParentPos().MovedBy(e.mouse.x , e.mouse.y);
41 
42  /// Only use dx and dy from the mouse event, because the relative mouse xy may change if our root moves b/c that moves us
43  /// and the relative mouse position given to us in mouse.x and mouse.y
44  if (e.type == EAGLE_EVENT_MOUSE_AXES) {
45  mdxtotal += e.mouse.dx;
46  mdytotal += e.mouse.dy;
47  }
48 
49  const int mdx = mdxtotal;
50  const int mdy = mdytotal;
51 
52  if (e.type == EAGLE_EVENT_MOUSE_AXES) {
53  /// TODO : FIXME : This doesn't work for the root gui
54  if (moving) {
55  Rectangle newarea = original_area.OuterArea();
56  newarea.MoveBy(mdx , mdy);
57  mwidget->SetWidgetArea(newarea , true);
58  }
59  else if (sizing) {
60  Rectangle oldarea = original_area.OuterArea();
61  Rectangle newarea = oldarea;
62  if (anchorw || anchorh) {
63  /// horizontal or vertical sizing
64  if (anchorw) {
65  /// horizontal sizing
66  if (size_corner == CELL_AREA_MIDDLELEFT) {
67  /// moving left edge
68  newarea.SetArea(oldarea.X() + mdx , oldarea.Y() , anchorw - mdx , oldarea.H());
69  }
70  if (size_corner == CELL_AREA_MIDDLERIGHT) {
71  /// moving right edge
72  newarea.SetArea(oldarea.X() , oldarea.Y() , anchorw + mdx , oldarea.H());
73  }
74  }
75  if (anchorh) {
76  /// vertical sizing
77  if (size_corner == CELL_AREA_TOPMIDDLE) {
78  /// moving top edge
79  newarea.SetArea(oldarea.X() , oldarea.Y() + mdy , oldarea.W() , anchorh - mdy);
80  }
81  else if (size_corner == CELL_AREA_BOTTOMMIDDLE) {
82  /// moving bottom edge
83  newarea.SetArea(oldarea.X() , oldarea.Y() , oldarea.W() , anchorh + mdy);
84  }
85  }
86  mwidget->SetWidgetArea(newarea , true);
87  }
88  else {
89  /// diagonal sizing
90  Pos2I newcorner = movept.MovedBy(mdx , mdy);
91  newarea.SetCorners(anchorpt.X() , anchorpt.Y() , newcorner.X() , newcorner.Y());
92  mwidget->SetWidgetArea(newarea , true);
93  }
94  }
95  else {
96  mwidget = 0;
97 
98  EAGLE_ASSERT(RootHandler());/// We need the root gui to tell us which widget is under the mouse
99 
100  WidgetBase* hwidget = RootHandler()?RootHandler()->GetWidgetAt(mabs.X() , mabs.Y()):(WidgetBase*)0;
101  if (blist.find(hwidget) == blist.end()) {
102  /// Not on blacklist
103  if (wlist.empty() || (wlist.find(hwidget) != wlist.end())) {
104  /// On whitelist, or no whitelist
105  if (hwidget &&
106  (hwidget->Flags().FlagOn(MOVEABLE) || hwidget->Flags().FlagOn(RESIZEABLE))) {
107  mwidget = hwidget;
108  original_area = mwidget->GetWidgetArea();
109  /// We need the absolute position here
110  abs_area = mwidget->AbsoluteArea();
111  }
112  }
113  }
114 
115  /// Set mouse icon here
116 
117  /// Icon depends on whether or not we are hovering over a margin cell
118  /// or border cell
119  MOUSE_POINTER_TYPE old_pointer_type = pointer_type;
120  pointer_type = POINTER_NORMAL;
121 
122  if (mwidget) {
123  CELL_AREA mcell = abs_area.OuterNP().GetCellArea(mabs.X() , mabs.Y());
124  CELL_AREA bcell = abs_area.BorderNP().GetCellArea(mabs.X() , mabs.Y());
125 
126  EAGLE_DEBUG(
127 /// EagleInfo() << "mcell = " << mcell << " , bcell = " << bcell << " , mabsxy = " << mabs.X() << "," << mabs.Y() << std::endl;
128 /// EagleInfo() << "area = " << abs_area << std::endl;
129  );
130 
131  if (mcell != CELL_AREA_OUTSIDE) {
132  if (mcell != CELL_AREA_MIDDLEMIDDLE) {
133  /// Hovering over a margin cell, show sizing mouse icon
134  const static MOUSE_POINTER_TYPE mptrs[9] = {
135  POINTER_NW,
136  POINTER_N,
137  POINTER_NE,
138  POINTER_W,
139  POINTER_NORMAL,
140  POINTER_E,
141  POINTER_SW,
142  POINTER_S,
143  POINTER_SE
144  };
145  if (sizing_enabled && mwidget->Flags().FlagOn(RESIZEABLE)) {
146  pointer_type = mptrs[mcell];
147  }
148  }
149  else {
150  /// We're inside the border area
151  if (bcell != CELL_AREA_MIDDLEMIDDLE) {
152  /// Hovering over a border cell, show grab mouse icon
153  if (moving_enabled && mwidget->Flags().FlagOn(MOVEABLE)) {
154  pointer_type = POINTER_GRAB;
155  }
156  }
157  }
158  }
159  }
160 
161  EagleGraphicsContext* win = RootHandler()?RootHandler()->GetDrawWindow():(EagleGraphicsContext*)0;
162 
163  EAGLE_ASSERT(win);/// We need the drawing window to set its mouse icon
164 
165  if (win && (pointer_type != old_pointer_type)) {
166  /// Pointer changed
167  if (pointer_type != POINTER_NORMAL) {
168  macquired = win->AcquireMousePointer(this , pointer_type , true);
169  EAGLE_ASSERT(macquired);
170  }
171  else {/// pointer_type == POINTER_NORMAL
172  if (macquired) {
173  win->ReleaseMousePointer(this);
174  macquired = false;
175  }
176  }
177  }
178  }
179  }
180  if (e.type == EAGLE_EVENT_MOUSE_BUTTON_UP) {
181  if (moving || sizing) {
182  if (e.mouse.button == 1) {
183  moving = false;
184  sizing = false;
185  if (macquired) {
186  EagleGraphicsContext* win = RootHandler()?RootHandler()->GetDrawWindow():(EagleGraphicsContext*)0;
187  if (win) {
188  win->ReleaseMousePointer(this);
189  }
190  macquired = false;
191  }
192  return DIALOG_INPUT_USED;
193  }
194  }
195  }
196  if ((e.type == EAGLE_EVENT_MOUSE_BUTTON_DOWN) && /// on mouse button
197  (mwidget) && /// don't try to move a null widget
198  (e.mouse.button == 1)) /// only if LMB is pressed
199  {
200  if (!moving && !sizing) {/// so we don't interrupt ourselves
201 // mxstart = e.mouse.x;
202 // mystart = e.mouse.y;
203  mdxtotal = 0;
204  mdytotal = 0;
205  original_area = mwidget->GetWidgetArea();
206  abs_area = mwidget->AbsoluteArea();
207  if (e.mouse.button == 1) {
208  if (mwidget) {/// If we are over a widget
209  if (moving_enabled && mwidget->Flags().FlagOn(MOVEABLE)) {
210  CELL_AREA bcell = abs_area.BorderNP().GetCellArea(mabs.X() , mabs.Y());
211  if ((bcell != CELL_AREA_OUTSIDE) && (bcell != CELL_AREA_MIDDLEMIDDLE)) {
212  /// User clicked on our border
213  moving = true;
214  /// If we have a drawing context, we can set the mouse pointer icon to the grabby hand to
215  /// indicate we're holding on to a widget
216  EagleGraphicsContext* win = RootHandler()?RootHandler()->GetDrawWindow():(EagleGraphicsContext*)0;
217  win->AcquireMousePointer(this , POINTER_GRABBED , true);
218  return DIALOG_INPUT_USED;
219  }
220  }
221  if (sizing_enabled && mwidget->Flags().FlagOn(RESIZEABLE)) {
222  CELL_AREA mcell = abs_area.OuterNP().GetCellArea(mabs.X() , mabs.Y());
223  if ((mcell != CELL_AREA_OUTSIDE) && (mcell != CELL_AREA_MIDDLEMIDDLE)) {
224  /// User clicked on our margin
225  size_corner = mcell;
226  sizing = true;
227  anchorw = 0;
228  anchorh = 0;
229  switch (mcell) {
230  case CELL_AREA_TOPLEFT :
231  movept.SetPos(original_area.OuterArea().X() , original_area.OuterArea().Y());
232  anchorpt.SetPos(original_area.OuterArea().BRX() , original_area.OuterArea().BRY());
233  break;
234  case CELL_AREA_TOPRIGHT :
235  movept.SetPos(original_area.OuterArea().BRX() , original_area.OuterArea().Y());
236  anchorpt.SetPos(original_area.OuterArea().X() , original_area.OuterArea().BRY());
237  break;
238  case CELL_AREA_BOTTOMLEFT :
239  movept.SetPos(original_area.OuterArea().X() , original_area.OuterArea().BRY());
240  anchorpt.SetPos(original_area.OuterArea().BRX() , original_area.OuterArea().Y());
241  break;
242  case CELL_AREA_BOTTOMRIGHT :
243  movept.SetPos(original_area.OuterArea().BRX() , original_area.OuterArea().BRY());
244  anchorpt.SetPos(original_area.OuterArea().X() , original_area.OuterArea().Y());
245  break;
246  case CELL_AREA_TOPMIDDLE :
247  anchorpt.SetPos(0 , original_area.OuterArea().BRY());
248  anchorh = original_area.OuterArea().H();
249  break;
250  case CELL_AREA_BOTTOMMIDDLE :
251  anchorpt.SetPos(0 , original_area.OuterArea().Y());
252  anchorh = original_area.OuterArea().H();
253  break;
254  case CELL_AREA_MIDDLELEFT :
255  anchorpt.SetPos(original_area.OuterArea().BRX() , 0);
256  anchorw = original_area.OuterArea().W();
257  break;
258  case CELL_AREA_MIDDLERIGHT :
259  anchorpt.SetPos(original_area.OuterArea().X() , 0);
260  anchorw = original_area.OuterArea().W();
261  break;
262  default:
263  break;
264  }
265  return DIALOG_INPUT_USED;
266  }
267  }
268  }
269  }
270  }
271  }
272  return DIALOG_OKAY;
273 }
