168 /// Right now only NAME , POS, DIM, and AREA are supported
169 void ApplyWidgetBaseAttributes(WidgetBase* widget , const AttributeValueMap& avmap) {
170 
171  if (!widget) {
172  throw EagleException("ApplyWidgetBaseAtributes : widget is NULL.\n");
173  }
174 
175  const ATTVALMAP& attribute_map = avmap.GetAttributeValueMap();
176 
177 /// widget = widget->GetDecoratorRoot();/// TODO : FIXME : ???
178 
179  map<string , string>::const_iterator cit = attribute_map.end();
180 
181  const char* cstr = 0;
182 
183  if ((cit = attribute_map.find("NAME")) != attribute_map.end()) {
184  widget->SetShortName(cit->second);
185  }
186 
187  if ((cit = attribute_map.find("POS")) != attribute_map.end()) {
188  cstr = cit->second.c_str();
189  int x = 0 , y = 0;
190  if (2 != sscanf(cstr , "%d,%d" , &x , &y)) {
191  throw EagleException(StringPrintF("ApplyWidgetBaseAtributes:: (POS) Failed to read x,y pair from (%s)\n" , cstr));
192  }
193  Rectangle warea = widget->OuterArea();
194  warea.SetPos(x,y);
195  widget->SetWidgetArea(warea);
196  }
197  if ((cit = attribute_map.find("DIM")) != attribute_map.end()) {
198  cstr = cit->second.c_str();
199  int w = 0 , h = 0;
200  if (2 != sscanf(cstr , "%d,%d" , &w , &h)) {
201  throw EagleException(StringPrintF("ApplyWidgetBaseAtributes:: (DIM) Failed to read w,h pair from (%s)\n" , cstr));
202  }
203  int x = widget->OuterArea().X();
204  int y = widget->OuterArea().Y();
205  if (w < 0 || h < 0) {
206  if (w < 0) {
207  w = abs(w);
208  x = x - w;
209  }
210  if (h < 0) {
211  h = abs(h);
212  y = y - h;
213  }
214  }
215  widget->SetWidgetArea(Rectangle(x,y,w,h));
216 // Pos(x,y);
217 // widget->SetWidgetDimensions(w,h);
218  }
219  if ((cit = attribute_map.find("AREA")) != attribute_map.end()) {
220  cstr = cit->second.c_str();
221  int x = 0 , y = 0 , w = 0 , h = 0;
222  if (4 != sscanf(cstr , "%d,%d,%d,%d" , &x , &y , &w , &h)) {
223  throw EagleException(StringPrintF("ApplyWidgetBaseAtributes:: (AREA) Failed to read x,y,w,h set from (%s)\n" , cstr));
224  }
225  widget->SetWidgetArea(Rectangle(x,y,w,h));
226  }
227 
228 }

232 void ApplyTextAttributes(WidgetBase* widget , const AttributeValueMap& avmap) {
233 
234  const ATTVALMAP& attribute_map = avmap.GetAttributeValueMap();
235 
236  BasicText* text_widget = dynamic_cast<BasicText*>(widget);
237 
238  if (!text_widget) {
239  throw EagleException(StringPrintF("ApplyTextAttributes - widget %p is not a BasicText widget!!!\n" , widget));
240  }
241 
242  map<string , string>::const_iterator cit = attribute_map.end();
243 
244  if ((cit = attribute_map.find("TEXT")) != attribute_map.end()) {
245  text_widget->SetupText(cit->second);
246  }
247  if ((cit = attribute_map.find("FONT")) != attribute_map.end()) {
248  text_widget->SetupText(text_widget->GetText() , GetFont(cit->second));
249  }
250  if ((cit = attribute_map.find("SHRINKWRAP")) != attribute_map.end()) {
251  if (strcmp(cit->second.c_str() , "YES") == 0) {
252  text_widget->ShrinkWrap();
253  }
254  }
255 }
