114 template <class WIDGET_TYPE>
115 WIDGET_TYPE* CreateWidget(std::string widget_class_name , std::string object_name , AttributeValueMap avmap , WIDGET_TYPE** pwidget_store) {
116  WIDGET_TYPE* pwidget = 0;
117  pwidget = dynamic_cast<WIDGET_TYPE*>(CreateWidgetObject(widget_class_name , object_name , avmap));
118  if (!pwidget) {
119  std::string avstr = avmap.ToString();
120  throw EagleException(StringPrintF("Failed to create %s class widget with specified parameters (%s)\n" ,
121  typeid(WIDGET_TYPE).name() , avstr.c_str()));
122  }
123  if (pwidget_store) {
124  *pwidget_store = pwidget;
125  }
126  return pwidget;
127 }
