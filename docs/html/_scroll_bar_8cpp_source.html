35 void BasicScrollBar::ResetScrollBarArea() {
36  Rectangle inner = InnerArea();
37  if (is_horizontal) {
38  Rectangle left_button_area = LayoutArea(inner , LayoutRectangle(0.0 , 0.0 , 0.1 , 1.0));
39  Rectangle right_button_area = LayoutArea(inner , LayoutRectangle(0.9 , 0.0 , 0.1 , 1.0));
40  Rectangle scroll_area = LayoutArea(inner , LayoutRectangle(0.1 , 0.0 , 0.8 , 1.0));
41  up_or_left_button->SetWidgetArea (left_button_area , false);
42  down_or_right_button->SetWidgetArea(right_button_area , false);
43  scroller->SetWidgetArea (scroll_area , false);
44  }
45  else {
46  Rectangle up_button_area = LayoutArea(inner , LayoutRectangle(0.0 , 0.0 , 1.0 , 0.1));
47  Rectangle down_button_area = LayoutArea(inner , LayoutRectangle(0.0 , 0.9 , 1.0 , 0.1));
48  Rectangle scroll_area = LayoutArea(inner , LayoutRectangle(0.0 , 0.1 , 1.0 , 0.8));
49  up_or_left_button->SetWidgetArea (up_button_area , false);
50  down_or_right_button->SetWidgetArea(down_button_area , false);
51  scroller->SetWidgetArea (scroll_area , false);
52  }
53  ResetHandleArea();
54 }

94 void BasicScrollBar::QueueUserMessage(const WidgetMsg& wmsg) {
95 
96  /// Messages from our scroll buttons and scroller
97  const WidgetMsg scroll_msg(scroller , TOPIC_SCROLL_WIDGET , SCROLL_VALUE_CHANGED);
98  const WidgetMsg ul_button_msg_click(up_or_left_button , TOPIC_BUTTON_WIDGET , BUTTON_CLICKED);
99  const WidgetMsg ul_button_msg_held(up_or_left_button , TOPIC_BUTTON_WIDGET , BUTTON_HELD);
100  const WidgetMsg dr_button_msg_click(down_or_right_button , TOPIC_BUTTON_WIDGET , BUTTON_CLICKED);
101  const WidgetMsg dr_button_msg_held(down_or_right_button , TOPIC_BUTTON_WIDGET , BUTTON_HELD);
102 
103  /// Scroll message that we send out
104  const WidgetMsg scrollbar_msg(this , TOPIC_SCROLL_WIDGET , SCROLL_VALUE_CHANGED);
105 
106  if ((wmsg == scroll_msg) ||
107  (wmsg == ul_button_msg_click) ||
108  (wmsg == ul_button_msg_held) ||
109  (wmsg == dr_button_msg_click) ||
110  (wmsg == dr_button_msg_held)) {
111  SetRedrawFlag();
112  RaiseEvent(scrollbar_msg);
113  return;
114  }
115  WidgetBase::QueueUserMessage(wmsg);
116 }
