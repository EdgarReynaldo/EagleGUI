215 bool MousePointerManager::AcquireMousePointer(void* caller , MOUSE_POINTER_TYPE ptype , bool use_system_pointer) {
216 
217  MousePointerSet* pset = pointer_set_in_use;// could be user or local custom pointer set
218  if (use_system_pointer) {
219  pset = &system_pointer_set;
220  }
221  MousePointer** new_pointer = &((*pset)[ptype]);
222  bool ret = SelectMousePointerDirect(new_pointer);
223 
224  if (ret) {
225  if (mouse_owners.back().first != caller) {
226  std::pair<void* , MousePointer**> mouse_owner(caller , new_pointer);
227  mouse_owners.push_back(mouse_owner);
228  }
229  current_pointer = new_pointer;
230  }
231  return ret;

236 void MousePointerManager::ReleaseMousePointer(const void* caller) {
237 
238  EAGLE_ASSERT(caller != this);// We never remove our own mouse pointer from the bottom of the deque
239  EAGLE_ASSERT(mouse_owners.size());// There should always be at least one mouse owner in the deque
240 
241  if (mouse_owners.back().first == caller) {
242  /// The pointer is in use, so we need to pop_back and then re-select, otherwise, just remove
243  mouse_owners.pop_back();
244  EAGLE_ASSERT(mouse_owners.size());
245  MousePointer** oldmp = mouse_owners.back().second;
246  bool oldmp_restored = SelectMousePointerDirect(oldmp);
247  EAGLE_ASSERT(oldmp_restored);
248  (void)oldmp_restored;
249  return;
250  }
251 
252  /// remove the topmost instance of the owner's mouse pointer, but only one so Acquire and Release pairs match up
253  std::list< std::pair<void* , MousePointer**> >::reverse_iterator it = mouse_owners.rbegin();
254  while (it != mouse_owners.rend()) {
255  if (caller == it->first) {
256  ///it = mouse_owners.erase(it);// no erase member that takes a reverse_iterator :PPP
