48 enum EAGLE_INIT_STATE {

50  EAGLE_SYSTEM = 1 << 0, ///< Install the system

52  EAGLE_FONTS = 1 << 2, ///< Support fonts
53  EAGLE_TTF_FONTS = 1 << 3, ///< Support ttf fonts
54  EAGLE_AUDIO = 1 << 4, ///< Support audio
55  EAGLE_SHADERS = 1 << 5, ///< Support shaders
56  EAGLE_PRIMITIVES = 1 << 6, ///< Support primitive drawing
57  EAGLE_KEYBOARD = 1 << 7, ///< Install the keyboard
58  EAGLE_MOUSE = 1 << 8, ///< Install the mouse
59  EAGLE_JOYSTICK = 1 << 9, ///< Install the joystick
60  EAGLE_TOUCH = 1 << 10,///< Install the touch driver
61 
62  EAGLE_STANDARD_INPUT = EAGLE_KEYBOARD | EAGLE_MOUSE | EAGLE_JOYSTICK,
63  EAGLE_STANDARD_SYSTEM = EAGLE_SYSTEM | EAGLE_IMAGES | EAGLE_FONTS | EAGLE_TTF_FONTS | EAGLE_AUDIO | EAGLE_PRIMITIVES,
64  EAGLE_STANDARD_SETUP = EAGLE_STANDARD_INPUT | EAGLE_STANDARD_SYSTEM,

67  EAGLE_FULL_SYSTEM = EAGLE_STANDARD_SYSTEM | EAGLE_SHADERS,
