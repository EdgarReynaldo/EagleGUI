85

90  unsigned int index = 0;
91 
92  while (index < stop) {

124  ++it;

132  break;
133  }
134  }

136  (paths[0].find_first_of(":") == std::string::npos)) {/// Windows drives use colon
137  relative = true;
138  }

140  if (!relative) {
141  return paths;
142  }
