420 void SelectText::FindCaretPos(int msx , int msy , int* pstrpos , int* plinenum) {
421  EAGLE_ASSERT(pstrpos);
422  EAGLE_ASSERT(plinenum);
423  Rectangle outer = OuterArea();
424  *pstrpos = -1;
425  *plinenum = -1;
426 
427  /// NOTE : This works for all HALIGN and VALIGN values
428 
429 /// TODO : GIMME THE WHAT FOR
430 /// EAGLE_ASSERT(lines.size());
431  if (!lines.size()) {return;}
432 
433  /// Allow margin space to be considered for selection purposes. It makes for nicer selection characteristics.
434  if (outer.Contains(msx,msy)) {
435 
436  if (msy < lineareas[0].Y()) {
437  *plinenum = 0;
438  }
439  else if (msy > lineareas[lineareas.size() - 1].BRY()) {
440  *plinenum = lineareas.size() - 1;
441  }
442  else {
443  for (int line = 0 ; line < nlines ; ++line) {
444  Rectangle r = lineareas[line];
445  int ty = r.Y();
446  int by = r.BRY();
447  if (msy >= ty && msy <= by) {
448  *plinenum = line;
449  }
450  }
451  }
452  std::string linestr = lines[*plinenum];
453  Rectangle r = lineareas[*plinenum];
454  int lx = r.X();
455  int rx = r.BRX();
456  if (msx < lx) {
457  *pstrpos = 0;
458  }
459  else if (msx >= rx) {
460  *pstrpos = (int)linestr.size();
461  }
462  else {
463  *pstrpos = 0;
464  for (int index = 0; index < (int)linestr.size() ; ++index) {
465  std::string sub = linestr.substr(0 , (unsigned int)index);
466  std::string sub2 = linestr.substr(0 , (unsigned int)index + 1);
467  float xpos = lx + text_font->Width(sub.c_str());
468  float xpos2 = lx + text_font->Width(sub2.c_str());
469  float cx = (xpos + xpos2) / 2.0;
470  if (msx >= xpos && msx < cx) {
471  *pstrpos = index;
472  break;
473  }
474  else if (msx >= cx && msx < xpos2) {
475  *pstrpos = index + 1;
476  break;
477  }
478  }
479  }
480  }
