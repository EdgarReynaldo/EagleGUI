55 bool CXX11Mutex::PrivateLockWaitFor(double timeout) {
56  EAGLE_ASSERT(Valid());
57 
58  if (!Timed()) {
59  throw EagleException("CXX11Mutex::PrivateLockWaitFor(double) : This is not a Timed() mutex!");
60  }
61 
62  int ms = int(1000.0*timeout);
63  if (ms < 1) {ms = 1;}
64  auto duration = std::chrono::milliseconds(ms);
65 
66  switch (type) {
67  case MTX_TIMED :
68  return MTX.mtx_timed->try_lock_for(duration);
69  break;
70  case MTX_RECURSIVE_TIMED :
71  return MTX.mtx_recursive_timed->try_lock_for(duration);
72  break;
73  default:
74  break;
75  }

141 bool CXX11Mutex::Create(bool multi_lockable , bool is_timed) {
142  Destroy();
143 
144  if (is_timed) {
145  /// Timed mutexes
146  if (multi_lockable) {
147  type = MTX_RECURSIVE_TIMED;
148  MTX.mtx_recursive_timed = new std::recursive_timed_mutex();
149  }
150  else {
151  type = MTX_TIMED;
152  MTX.mtx_timed = new std::timed_mutex();
153  }
154  }
155  else {
156  /// Non-timed mutexes
157  if (multi_lockable) {
158  type = MTX_RECURSIVE;
159  MTX.mtx_recursive = new std::recursive_mutex();
160  }
161  else {
162  type = MTX_PLAIN;
163  MTX.mtx = new std::mutex();
164  }
165  }
166  return true;
167 }
