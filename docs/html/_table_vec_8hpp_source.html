68 void TableVec2D<TYPE>::Resize(unsigned int numcols , unsigned int numrows , TYPE t) {
69  int dcols = numcols - ncols;
70  int drows = numrows - nrows;
71 
72  if (dcols == 0 && drows == 0) {return;}/// Nothing to do

74  /// Empty table

76  table.resize(drows , std::vector<TYPE>(dcols , t));
77  ncols = dcols;
78  nrows = drows;

85  table.clear();

87  return;
88  }

96  table.resize(numrows , std::vector<TYPE>(ncols , t));
