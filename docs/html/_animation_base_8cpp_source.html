200 /*! \brief Sets the elapsed percentage of the animation.
201  *
202  * @param percent Equal to animation time divided by loop duration.
203  *
204  * AnimationBase::LoopComplete will be called for each loop completed
205  * when advancing the animation time.
206  * AnimationBase::Complete will be called when the animation is complete.
207  *
208  * Each function will emit an event. Use an EagleEventQueue or
209  * EagleEventListener to listen for events from this object.
210  *
211  * event_queue.ListenTo(&animation_object);
212  *
213  * @retval None This function cannot fail.
214  */
215 
216 void AnimationBase::SetAnimationPercent(double percent) {
217 
218  double old_animation_percent = animation_percent;
219  animation_percent = percent;
220  normal_percent = GetNormalizedPercent();
221 
222  int old_loop_num = (int)old_animation_percent;
223  loop_num = (int)animation_percent;
224  frame_num = (int)(nframes*(normal_percent));
225 
226  OnSetAnimationPercent();
227 
228  for (int i = old_loop_num ; i < loop_num ; ++i) {
229  if (PingPong()) {
230  if (i % 2 == 0) {
231  LoopComplete();
232  }
233  }
234  else {
235  LoopComplete();
236  }
237  }
238  if (animation_percent >= nloops && old_animation_percent < nloops) {
239  Complete();
240  }
241 }

253 /*! \brief Returns the normalized animation percent from [0.0 to 1.0). */
254 
255 double AnimationBase::GetNormalizedPercent() {
256  double pct = GetAnimationPercent();
257  bool forward = true;
258  if (PingPong()) {
259  pct = fmod(pct , 2.0);
260  if (pct >= 1.0) {
261  pct -= 1.0;
262  forward = false;
263  }
264  }
265  else if (Backwards()) {
266  forward = false;
267  }
268  else {
269  pct = fmod(pct , 1.0);
270  }
271  if (!forward) {
272  return 1.0 - pct;
273  }
274  return pct;
275 }
