69 int BasicScroller::PrivateHandleEvent(EagleEvent e) {
70  int retmsg = DIALOG_OKAY;
71  if (e.type == EAGLE_EVENT_MOUSE_BUTTON_DOWN && e.mouse.button == 1) {
72  int msx = e.mouse.x;
73  int msy = e.mouse.y;
74  if (InnerArea().Contains(msx,msy)) {
75  retmsg |= DIALOG_TAKE_FOCUS | DIALOG_INPUT_USED;
76  if (scroll_handle_area.Contains(msx,msy)) {
77  mouse_drag_handle = true;
78  start_mouse_x = msx;
79  start_mouse_y = msy;
80  start_percent = scroll_percent;
81  }
82  else {
83  down = true;
84  repeat_elapsed = repeat_previous = 0.0;
85  float percent_to_scroll_by = ((float)length_in_view/total_length)/2.0f;
86  if (horizontal_scroller) {
87  if (msx < scroll_handle_area.X()) {
88  /// Scroll left, by 50% of the view length
89  ScrollBy(-percent_to_scroll_by);
90  repeat_scroll_increment = -length_in_view/2;
91  }
92  else if (msx > scroll_handle_area.BRX()) {
93  /// Scroll right, by 50% of the view length
94  ScrollBy(percent_to_scroll_by);
95  repeat_scroll_increment = length_in_view/2;
96  }
97  }
98  else {/// Vertical scroller
99  if (msy < scroll_handle_area.Y()) {
100  /// Scroll up
101  ScrollBy(-percent_to_scroll_by);
102  repeat_scroll_increment = -length_in_view/2;
103  }
104  else if (msy > scroll_handle_area.BRY()) {
105  ScrollBy(percent_to_scroll_by);
106  repeat_scroll_increment = length_in_view/2;
107  }
108  }
109  }
110  }
111  }
112  if (e.type == EAGLE_EVENT_MOUSE_AXES) {
113  if (mouse_drag_handle) {
114  int msx = e.mouse.x;
115  int msy = e.mouse.y;
116  int dx = msx - start_mouse_x;
117  int dy = msy - start_mouse_y;
118  int delta = horizontal_scroller?dx:dy;
119  int length = 0;
120  if (horizontal_scroller) {
121  length = (InnerArea().W() - scroll_handle_area.W());
122  }
123  else if (!horizontal_scroller) {
124  length = (InnerArea().H() - scroll_handle_area.H());
125  }
126  if (length == 0) {length = 1;}
127  float percent = (float)delta/length;
128  SetScrollPercent(start_percent + percent);
129  }
130  }
131  if (e.type == EAGLE_EVENT_MOUSE_BUTTON_UP) {
132  if (e.mouse.button == 1) {
133  down = false;
134  if (mouse_drag_handle) {
135  mouse_drag_handle = false;
136  retmsg |= DIALOG_INPUT_USED;
137  }
138  }
139  }
140  return retmsg;
141 }

169 int BasicScroller::PrivateUpdate(double tsec) {
170  (void)tsec;
171  if (down) {
172  repeat_elapsed += tsec;
173  if (repeat_elapsed >= repeat_delay) {
174  if (repeat_previous < repeat_delay) {
175  repeat_previous = repeat_delay;
176  }
177  double dt = repeat_elapsed - repeat_previous;
178  int num_repeat = dt/repeat_rate;
179  for (int i = 0 ; i < num_repeat ; ++i) {
180  ScrollBy(repeat_scroll_increment);
181 /// RaiseEvent(WidgetMsg(this , TOPIC_SCROLL_WIDGET , SCROLL_VALUE_CHANGED));
182  }
183  repeat_previous += (double)num_repeat*repeat_rate;
184  EAGLE_ASSERT(repeat_previous <= repeat_elapsed);
185  }
186  }
187  return DIALOG_OKAY;
188 }
