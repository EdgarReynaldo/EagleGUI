131 int BasicButton::PrivateCheckInputs() {
132 
133  unsigned int retmsg = DIALOG_OKAY;
134  bool activated = false;
135  bool released = false;
136 
137  int msx = mouse_x;
138  int msy = mouse_y;
139 
140  /// Track whether the button is still being held since the last activation
141  if (user_activated || focuskey_activated || pointer_activated) {
142  /// Look for input releases
143  if (user_activated) {
144  Input click = input_group.FindClickInput();
145  if (click != input_key_press(EAGLE_KEY_NONE)) {
146  Input release(click.Source() , RELEASE , click.Value());
147  if (release) {
148  released = true;
149  }
150  }
151  }
152  if (focuskey_activated) {
153  if (input_key_release(EAGLE_KEY_SPACE) || input_key_release(EAGLE_KEY_ENTER)) {released = true;}
154  }
155  if (pointer_activated) {
156  if (input_mouse_release(LMB)) {released = true;}
157  }
158  if (released) {
159  SetButtonState(Flags().FlagOn(HOVER) , true);
160  user_activated = false;
161  focuskey_activated = false;
162  pointer_activated = false;
163  } else {/// HELD
164  down_time_left = spring_duration;
165 /// RaiseEvent(WidgetMsg(this , TOPIC_BUTTON_WIDGET , BUTTON_HELD));/// Don't send HELD message - it just spams everything
166  }
167  return DIALOG_OKAY;
168  }
169 
170 
171  if (input_group) {
172 
173  EagleInfo() << "input_group active" << std::endl;
174  activated = true;
175  if (btn_action_type == SPRING_BTN) {user_activated = true;}
176 
177  } else if (Flags().FlagOn(HASFOCUS) && (input_key_press(EAGLE_KEY_SPACE) || input_key_press(EAGLE_KEY_ENTER))) {
178 
179  activated = true;
180  if (btn_action_type == SPRING_BTN) {focuskey_activated = true;}
181 
182  } else if (input_mouse_press(LMB)) {
183  Rectangle real = OuterArea();
184  click_area->MoveBy(real.X() , real.Y());
185 
186  EAGLE_DEBUG(
187  EagleInfo() << "Mouse = " << msx << "," << msy << " Real = " << real << std::endl;
188  );
189 
190  if (real.Contains(msx,msy)) {
191  activated = true;
192  /// If there is no click area we use the whole button
193  if (click_area) {
194  if (!(click_area->Contains(msx,msy))) {
195  activated = false;
196  }
197  }
198  if (activated) {
199  EagleInfo() << "Click area activated." << std::endl;
200  if (btn_action_type == SPRING_BTN) {pointer_activated = true;}
201  }
202  }
203  click_area->MoveBy(-real.X() , -real.Y());
204  }
205  just_activated = false;
206  if (activated) {// click or key press has activated the button action
207  just_activated = true;
208  retmsg |= DIALOG_TAKE_FOCUS;
209  retmsg |= DIALOG_INPUT_USED;
210  bool up = btn_state%2 == 0;
211  switch (btn_action_type) {
212  case SPRING_BTN :
213  if (up) {
214  up = false;
215  }
216  break;
217  case TOGGLE_BTN :
218  up = !up;
219  break;
220  default : throw EagleException("BasicButton::PrivateCheckInputs - btn_action_type unknown");break;
221  }
222  SetButtonState(Flags().FlagOn(HOVER) , up);
223  if (Flags().FlagOn(ALLOW_CLOSE)) {
224  retmsg |= DIALOG_CLOSE;
225  }
226  }
227  return retmsg;
228 
229 }

265 int BasicButton::PrivateUpdate(double tsec) {
266  if (btn_action_type == SPRING_BTN) {
267  bool up = (btn_state%2) == 0;
268  bool hover = Flags().FlagOn(HOVER);
269 
270  if (!up) {
271  repeat_elapsed += tsec;
272  if (repeat_elapsed >= repeat_delay) {
273  if (repeat_previous < repeat_delay) {
274  repeat_previous = repeat_delay;
275  }
276  double dt = repeat_elapsed - repeat_previous;
277  int num_repeat = dt/repeat_rate;
278  for (int i = 0 ; i < num_repeat ; ++i) {
279  RaiseEvent(WidgetMsg(this , TOPIC_BUTTON_WIDGET , BUTTON_HELD));
280  }
281  repeat_previous += (double)num_repeat*repeat_rate;
282  EAGLE_ASSERT(repeat_previous <= repeat_elapsed);
283  }
284  }
285 
286  if (!up && !user_activated && !focuskey_activated && !pointer_activated) {
287  down_time_left -= tsec;
288  if (down_time_left <= 0.0) {
289  down_time_left = spring_duration;
290  up = true;
291  SetButtonState(hover , up);
292  }
293  }
294  }
295  return DIALOG_OKAY;
296 }
