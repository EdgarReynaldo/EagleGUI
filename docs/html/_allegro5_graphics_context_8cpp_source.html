692 void Allegro5GraphicsContext::DrawVTextString(EagleFont* font , std::string str , float x , float y , EagleColor c ,
693  HALIGNMENT halign , VALIGNMENT valign , int letter_spacing) {
694  if (!str.size()) {return;}
695  EAGLE_ASSERT(font);
696  EAGLE_ASSERT(font->Valid());
697  Allegro5Font* a5font = dynamic_cast<Allegro5Font*>(font);
698  EAGLE_ASSERT(a5font);
699  ALLEGRO_FONT* f = a5font->AllegroFont();
700  EAGLE_ASSERT(f);
701 
702  if (letter_spacing < 0) {
703  letter_spacing = al_get_font_line_height(f)/8;
704  }
705 
706  int vheight = font->VHeight(str , -1);
707  int vwidth = font->VWidth(str , 0);
708 
709  switch (valign) {
710  case VALIGN_CENTER :
711  y -= vheight/2;
712  break;
713  case VALIGN_BOTTOM :
714  y -= vheight;
715  break;
716  default : break;
717  };
718  al_hold_bitmap_drawing(true);
719  for (unsigned int i = 0 ; i < str.size() ; ++i) {
720  int bbx = 0 , bby = 0 , bbw = 0 , bbh = 0;
721  ALLEGRO_GLYPH g;
722  memset(&g , 0 , sizeof(g));
723  if (al_get_glyph(f , 0 , str[i] , &g)) {
724  int ox = 0;
725  switch (halign) {
726  case HALIGN_LEFT :
727  break;
728  case HALIGN_CENTER :
729  ox += (vwidth - g.w)/2;
730  break;
731  case HALIGN_RIGHT :
732  ox += vwidth - g.w;
733  break;
734  default : break;
735  };
736  al_draw_bitmap_region(g.bitmap , g.x , g.y , g.w , g.h , x + ox , y , 0);
737  y += g.h + letter_spacing;
738  }
739  }
740  al_hold_bitmap_drawing(false);
741 }
