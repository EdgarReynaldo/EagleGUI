28 void DefaultCBDrawFunc(EagleGraphicsContext* win , Rectangle area , int xpos , int ypos , const WidgetColorset& wc , bool up , bool hover) {
29  /// BG color
30  area.MoveBy(xpos,ypos);
31  win->DrawFilledRectangle(area , wc[BGCOL]);
32 
33  /// Shadow box
34  area.Shrink(2);
35  win->DrawFilledRectangle(area , wc[SDCOL]);
36 
37  /// FG box
38  area.Shrink(4);
39  if (up) {
40  area.MoveBy(-2,-2);
41  }
42  win->DrawFilledRectangle(area , hover?wc[HVRCOL]:wc[FGCOL]);
43 
44  if (!up) {
45  Pos2D p1(area.W() , area.H());
46  Pos2D pcenter = p1*0.5;
47  Pos2D left(p1.X()/4.0 , pcenter.Y());
48  Pos2D bottom(pcenter.X() , p1.Y()*0.75);
49  Pos2D topright(p1.X()*0.95 , p1.Y()*0.05);
50 
51  left.MoveBy(area.X() , area.Y());
52  bottom.MoveBy(area.X() , area.Y());
53  topright.MoveBy(area.X() , area.Y());
54 
55  float thickness = 0.08*(p1.X()>p1.Y()?p1.X():p1.Y());
56  win->DrawLine(left.X() , left.Y() , bottom.X() , bottom.Y() , thickness , wc[MGCOL]);
57  win->DrawLine(topright.X() , topright.Y() , bottom.X() , bottom.Y() , thickness , wc[MGCOL]);
58  }
59 }

63 void BasicCheckBox::RefreshImages(EagleGraphicsContext* win) {
64  FreeImages();
65  if (draw_func) {
66  for (int i = 0 ; i < 4 ; ++i) {
67  bool up = (i%2)==0;
68  bool hover = i/2;
69  EagleImage* img = win->CreateImage(OuterArea().W() , OuterArea().H());
70  win->PushDrawingTarget(img);
71  draw_func(win , Rectangle(0 , 0 , img->W() , img->H()) , 0 , 0 , WidgetColors() , up , hover);
72  cb_images[i] = img;
73  win->PopDrawingTarget();
74  IconButton::SetImages(cb_images[0] , cb_images[1] , cb_images[2] , cb_images[3]);
75  }
76  }
77  current = true;
78 }
