123 void* A5InputThreadProcess(EagleThread* thread , void* input_handler) {
124  EagleInputHandler* handler = (EagleInputHandler*)input_handler;
125  Allegro5InputHandler* a5_input_handler = dynamic_cast<Allegro5InputHandler*>(handler);
126 
127  if (!a5_input_handler) {
128  throw EagleException("A5InputThreadProcess : Input handler is invalid.");
129  }
130 
131  while (!thread->ShouldStop()) {
132  ALLEGRO_EVENT ev;
133  al_wait_for_event(a5_input_handler->input_queue , &ev);
134 
135  if (ev.any.source == &a5_input_handler->input_extra_event_source &&
136  ev.type == EAGLE_EVENT_USER_START &&
137  A5INPUT_THREAD_SHOULD_STOP == (A5INPUT_THREAD_MESSAGE)ev.user.data1) {
138  break;
139  }
140  EagleEvent ee = GetEagleInputEvent(ev);
141 
142  ee.window = Eagle::EagleLibrary::System("Allegro5")->GetWindowManager()->GetActiveWindow();///Allegro5GraphicsContext::GetActiveWindow();
143  ee.source = a5_input_handler;
144 
145  EagleInfo() << "Input Event #" << ee.type << " received." << std::endl;
146 
147  if (IsKeyboardEvent(ee)) {
148  a5_input_handler->keyboard_event_handler.RespondToEvent(ee , thread);
149  a5_input_handler->keyboard_event_handler.TakeNextEvent(thread);/// discard event after emitting
150  }
151  else if (IsMouseEvent(ee)) {
152  a5_input_handler->mouse_event_handler.RespondToEvent(ee , thread);
153  a5_input_handler->mouse_event_handler.TakeNextEvent(thread);/// discard event after emitting
154  }
155  else if (IsTouchEvent(ee)) {
156  a5_input_handler->touch_event_handler.RespondToEvent(ee , thread);
157  a5_input_handler->touch_event_handler.TakeNextEvent(thread);/// discard event after emitting
158  }
159  else if (IsJoystickEvent(ee)) {
160  a5_input_handler->joystick_event_handler.RespondToEvent(ee , thread);
161  a5_input_handler->joystick_event_handler.TakeNextEvent(thread);/// discard event after emitting
162  }
163  }
164  return input_handler;
165 }
