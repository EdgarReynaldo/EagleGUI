225 void TextDecorator::SetHoverState(bool state) {
226 
227  if (!state) {
228  basic_text_widget->SetHoverState(false);
229  text_widget_layout->SetHoverState(false);
230  WidgetDecorator::SetHoverState(false);
231  return;
232  }
233 
234  WidgetHandler* wh = NearestParentGui();
235  if (wh) {
236  int mx = wh->GetMouseX();
237  int my = wh->GetMouseY();
238 
239  Rectangle r;
240 
241  r = basic_text_widget->OuterArea();
242  if (r.Contains(mx,my)) {
243  basic_text_widget->SetHoverState(true);
244  }
245 
246  r = text_widget_layout->OuterArea();
247  if (r.Contains(mx,my)) {
248  text_widget_layout->SetHoverState(true);
249  }
250 
251  r = OuterArea();
252  if (r.Contains(mx,my)) {
253  WidgetDecorator::SetHoverState(true);
254  }
255  }
256  else {
257  basic_text_widget->SetHoverState(true);
258  text_widget_layout->SetHoverState(true);
259  WidgetDecorator::SetHoverState(true);
260  }
261 }
