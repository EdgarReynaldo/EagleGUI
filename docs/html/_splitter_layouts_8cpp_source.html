114 int TwoWaySplitter::PrivateHandleEvent(EagleEvent e) {
115  int ret = DIALOG_OKAY;
116 
117 
118  if (IsMouseEvent(e)) {
119  int mx = e.mouse.x;
120  int my = e.mouse.y;
121  olddrag = drag;
122  bool hover = false;
123  if ( (hover = GetHandleArea().Contains(mx,my)) ) {
124  // hover over handle
125  /// TODO : ShowMouseIcon
126  if (e.type == EAGLE_EVENT_MOUSE_BUTTON_DOWN && e.mouse.button == 1) {
127  /// click
128  drag = true;
129  dragxstart = mx;
130  dragystart = my;
131  divposstart = divider_position;
132  ret |= DIALOG_INPUT_USED;
133  }
134  }
135 
136  if (e.type == EAGLE_EVENT_MOUSE_BUTTON_UP && e.mouse.button == 1) {
137  drag = false;
138  }
139 
140  if (mouse_window) {// only if we have a mouse window
141 
142  MOUSE_POINTER_TYPE POINTER_IN_USE = POINTER_NS;// up-down arrow
143  if (splitter_type == SPLITTER_VERTICAL) {
144  POINTER_IN_USE = POINTER_EW;// left-right arrow
145  }
146 
147  if (hover && !oldhover) {
148  // just got mouse back over us
149  mouse_window->AcquireMousePointer(this , POINTER_IN_USE , true);
150  if (!drag) {
151  // we aren't dragging, so this is the first time we have gotten hover, so take the mouse
152  // mouse_window->AcquireMousePointer(this , POINTER_IN_USE , true);
153  }
154  else {
155  // drag is in effect, so just leave the mouse pointer alone
156  }
157  }
158 
159  if (drag && !olddrag) {
160  mouse_window->AcquireMousePointer(this , POINTER_IN_USE , true);
161  }
162 
163  if (!hover && oldhover) {
164  // just lost hover
165  mouse_window->ReleaseMousePointer(this);
166  }
167  if (!drag && olddrag) {
168  mouse_window->ReleaseMousePointer(this);
169  }
170  }
171 
172  if (drag && e.type == EAGLE_EVENT_MOUSE_AXES) {
173  int dx = mx - dragxstart;
174  int dy = my - dragystart;
175  int newdivpos = divposstart;
176  switch (splitter_type) {
177  case SPLITTER_VERTICAL:
178  newdivpos += dx;
179  break;
180  case SPLITTER_HORIZONTAL:
181  newdivpos += dy;
182  break;
183  }
184  SetDividerPos(newdivpos);
185  }
186 
187  oldhover = hover;
188  olddrag = drag;
189 
190  }
191  return ret;
192 }

257 Rectangle TwoWaySplitter::RequestWidgetArea(int widget_slot , int newx , int newy , int newwidth , int newheight) const {
258 
259  (void)newx;
260  (void)newy;
261  (void)newwidth;
262  (void)newheight;
263 
264  if ((widget_slot != 0) && (widget_slot != 1)) {
265  throw EagleException(StringPrintF("ERROR : TwoWaySplitter::PlaceWidget - Slot %d is invalid." , widget_slot));
266  }
267 
268  const WidgetBase* widget = GetWidget(widget_slot);
269 
270  if (!widget) {
271  return Rectangle(-1,-1,-1,-1);
272  }
273 
274  Rectangle r = InnerArea();
275  int x = r.X();
276  int y = r.Y();
277  int w = r.W();
278  int h = r.H();
279  switch (splitter_type) {
280  case SPLITTER_VERTICAL :
281  if (widget_slot == 0) {
282  // left panel
283  x += 0;
284  w = divider_position;
285  }
286  else if (widget_slot == 1) {
287  // right panel
288  x += divider_position + divider_size;
289  w = (r.W() - (divider_position + divider_size));
290  }
291  break;
292  case SPLITTER_HORIZONTAL :
293  if (widget_slot == 0) {
294  // top panel
295  y += 0;
296  h = divider_position;
297  }
298  else if (widget_slot == 1) {
299  // bottom panel
300  y += divider_position + divider_size;
301  h = (r.H() - (divider_position + divider_size));
302  }
303  break;
304  }
305 
306  return Rectangle(x,y,w,h);
307 }
