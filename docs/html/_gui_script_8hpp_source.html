23  *
24  * Specification for an EagleGuiScript file (.egs). File is text, with declarations for EagleColor's, WidgetColorsets, and Widgets
25  *
26  * Currently, there are two sections available in an .egs script. There is a "Declarations" section and a "Functions" section.
27  *
28  * Declarations follow the format of a class name followed by whitespace then by the object name, optional whitespace,
29  * and an equals sign. The following lines consist of attribute name and value pairs separated by a colon.
30  * Each line is treated as a set of attribute value pairs. You may separate these pairs by a newline or by a semi colon,
31  * but they must be kept separate. Atrributes and values may be surrounded by whitespace, which will be trimmed.
32  * A newline by itself signals the end of the object definition. Generally, value strings must not contain whitespace.
33  * Anything after an equals sign and before a separate newline will be parsed as attribute value pair sets.
34  *
35  * Supported class names include EagleColor, WidgetColorset, or any Widget class name
36  *
37  * New section named "Commands" which includes commands to create relationships between widgets
38  *
39  * [Declarations]
40  *
41  * CLASSNAME OBJECTNAME = ATTRIBUTE:VALUE ; ATTRIBUTE:VALUE;
42  * ATTRIBUTE : VALUE
43  *
44  * CLASSNAME OBJECTNAME =
45  * ATTRIBUTE : VALUE ; ATTRIBUTE : VALUE
46  *
47  * EagleColor white = RGBA(255,255,255,255)
48  *
49  * WidgetColorset colorset1 =
50  * SDCOL : RGBA(0,0,0,255)
51  * HLCOL : FRGBA(1.0,1.0,1.0,1.0)
52  * TXTCOL : white
53  *
54  * [Commands]
55  * LAYOUT_CHILDREN root_layout = {abc,def,ghi,jkl,mno,pqr,stu,vwx,yz}
56  * GUI_LAYOUT gui = root_layout
57  *
58  * LAYOUT_CHILD_RELATIVE root_layout = widget:%f,%f,%f,%f
59  *
60  * METHOD OBJECT = ARGUMENTS
61  *
62  * [Functions] TODO : Implment GuiScript2 scriptability
63  */
