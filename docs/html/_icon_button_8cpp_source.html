79 void IconButton::DisplayIcon(EagleGraphicsContext* win , BUTTON_STATE state , int xpos , int ypos , EagleColor tint) {
80  EagleImage* image = original_images[(int)state];
81  EAGLE_ASSERT(image && image->Valid());
82 
83  Transform t = win->GetTransformer()->GetViewTransform();
84 
85  t.Scale((double)InnerArea().W()/image->W() , (double)InnerArea().H()/image->H());
86  t.Translate(InnerArea().X() + xpos , InnerArea().Y() + ypos);
87 
88  win->GetTransformer()->PushViewTransform(t);
89 
90  win->DrawTinted(image , 0 , 0 , tint);

92  win->GetTransformer()->PopViewTransform();
