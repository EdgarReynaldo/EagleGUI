44 void GuiButton::ResetRadii() {
45  // Uses the largest size shape that will fit into the rectangle
46  Rectangle r = InnerArea();
47  switch (btn_shape) {
48  case RECTANGLE_BTN :
49  break;
50  case CIRCLE_BTN :
51  rad_a = (r.W()<r.H())?(r.W()/2):(r.H()/2);
52  break;
53  case ROUNDED_BTN :
54  {
55  rad_a = hradpercent*InnerArea().W()/2.0;
56  rad_b = vradpercent*InnerArea().H()/2.0;
57  }
58  break;
59  case ELLIPSE_BTN :
60  rad_a = r.W()/2;
61  rad_b = r.H()/2;
62  break;
63  }
64  SetBgRedrawFlag();
65 }

68 void GuiButton::ResetClickArea() {
69  switch(btn_shape) {
70  case RECTANGLE_BTN :
71  SetClickArea(new Rectangle(0 , 0 , InnerArea().W() , InnerArea().H()) , true);
72  break;
73  case ROUNDED_BTN :
74  SetClickArea(new RoundedRectangle(Rectangle(0 , 0 , InnerArea().W() , InnerArea().H()) , rad_a , rad_b) , true);
75  break;
76  case ELLIPSE_BTN :
77  SetClickArea(new Ellipse(0 , 0 , InnerArea().W() , InnerArea().H()) , true);
78  break;
79  case CIRCLE_BTN :
80  SetClickArea(new Circle(InnerArea().W()/2 , InnerArea().H()/2 , rad_a) , true);
81  break;
82  default :
83  break;
84  }

243 void DrawGuiButtonShape(EagleGraphicsContext* win , GuiButton* btn , int x , int y) {
244  if (!btn && !win) {
245  throw EagleException("Null win and Null btn passed to DrawGuiButtonShape.\n");
246  }
247  else if (!btn) {
248  throw EagleException("Null btn passed to DrawGuiButtonShape.\n");
249  }
250  else if (!win) {
251  throw EagleException("Null win passed to DrawGuiButtonShape.\n");
252  }
253  bool up = (btn->ButtonState()%2) == 0;
254  int rad_a = btn->RadiusA();
255  int rad_b = btn->RadiusB();
256 /// int rad_b = btn->RadiusB();
257  Rectangle r = btn->InnerArea();
258  r.MoveBy(x,y);
259  WidgetColorset wc = btn->WidgetColors();
260  /// TODO : Add hover coloring?
261  if (btn->Flags().FlagOn(HOVER)) {
262  switch(btn->ButtonShape()) {
263  case RECTANGLE_BTN :
264  up?
265  r.DrawGuiRectUp(win , wc[HVRCOL] , wc[SDCOL]):
266  r.DrawGuiRectDown(win , wc[MGCOL] , wc[SDCOL]);
267  break;
268  case CIRCLE_BTN :
269  up?
270  r.DrawGuiCircleUp(win , rad_a , wc[HVRCOL] , wc[SDCOL]):
271  r.DrawGuiCircleDown(win , rad_a , wc[MGCOL] , wc[SDCOL]);
272  break;
273  case ROUNDED_BTN :
274  up?
275  r.DrawGuiRoundedUp(win , rad_a , rad_b , wc[HVRCOL] , wc[SDCOL]):
276  r.DrawGuiRoundedDown(win , rad_a , rad_b , wc[MGCOL] , wc[SDCOL]);
277  break;
278  case ELLIPSE_BTN :
279  up?
280  r.DrawGuiEllipseUp(win , wc[HVRCOL] , wc[SDCOL]):
281  r.DrawGuiEllipseDown(win , wc[MGCOL] , wc[SDCOL]);
282  break;
283  }
284  }
285  else {
286  switch(btn->ButtonShape()) {
287  case RECTANGLE_BTN :
288  up?
289  r.DrawGuiRectUp(win , wc[FGCOL] , wc[MGCOL]):
290  r.DrawGuiRectDown(win , wc[MGCOL] , wc[BGCOL]);
291  break;
292  case CIRCLE_BTN :
293  up?
294  r.DrawGuiCircleUp(win , rad_a , wc[FGCOL] , wc[MGCOL]):
295  r.DrawGuiCircleDown(win , rad_a , wc[MGCOL] , wc[BGCOL]);
296  break;
297  case ROUNDED_BTN :
298  up?
299  r.DrawGuiRoundedUp(win , rad_a , rad_b , wc[FGCOL] , wc[MGCOL]):
300  r.DrawGuiRoundedDown(win , rad_a , rad_b , wc[MGCOL] , wc[BGCOL]);
301  break;
302  case ELLIPSE_BTN :
303  up?
304  r.DrawGuiEllipseUp(win , wc[FGCOL] , wc[MGCOL]):
305  r.DrawGuiEllipseDown(win , wc[MGCOL] , wc[BGCOL]);
306  break;
307  }
308  }
309 }

313 void DrawGuiButtonText(EagleGraphicsContext* win , GuiButton* btn , int x , int y) {
314  if (!btn && !win) {
315  throw EagleException("Null win and Null btn passed to DrawGuiButtonText.\n");
316  }
317  else if (!btn) {
318  throw EagleException("Null btn passed to DrawGuiButtonText.\n");
319  }
320  else if (!win) {
321  throw EagleException("Null win passed to DrawGuiButtonText.\n");
322  }
323 
324  EagleFont* text_font = btn->Font();
325  string text = btn->Text();
326 
327  if (text.length()) {
328  EAGLE_ASSERT(text_font);
329  if (!text_font) {
330  throw EagleException(StringPrintF("GuiButton object has text (\"%s\" but no associated font has been set!!!\n",text.c_str()));
331  }
332  Rectangle r = btn->InnerArea();
333  int tx = x + r.X() + r.W()/2;
334  int ty = y + r.Y() + r.H()/2;
335 
336  win->DrawGuiTextString(text_font , text.c_str() , tx , ty , btn->GetColor(TXTCOL) , HALIGN_CENTER , VALIGN_CENTER);
