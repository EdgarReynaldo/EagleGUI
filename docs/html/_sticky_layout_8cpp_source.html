160 int StickyLayout::PrivateHandleEvent(EagleEvent ee) {
161  if (ee.type == EAGLE_EVENT_WIDGET) {
162  if (ee.widget.topic == TOPIC_DIALOG && ee.widget.msgs == DIALOG_I_MOVED) {
163  WidgetBase* from = ee.widget.from;
164  for (int i = 0 ; i < (int)anchors.size() ; ++i) {
165  WidgetBase* awidget = anchors[i].AnchorWidget();
166  if (awidget == from) {
167  RepositionChild(i);
168  }
169  }
170  }
171  }
172  return DIALOG_OKAY;
173 }

185 Rectangle StickyLayout::RequestWidgetArea(int widget_slot , int newx , int newy , int newwidth , int newheight) {
186 
187  const WidgetBase* widget = GetWidget(widget_slot);
188 
189  if (!widget) {
190  return BADRECTANGLE;
191  }
192 
193  Rectangle r = widget->OuterArea();
194  (void)newx;
195  (void)newy;
196 
197  /// newx and newy are ignored by this layout
198  if (newwidth == INT_MAX) {newwidth = r.W();}
199  if (newheight == INT_MAX) {newheight = r.H();}
200 
201  StickyPosition sp = anchors[widget_slot];
202  Pos2I anchor = sp.GetAnchorPoint();
203  HALIGNMENT halign2 = sp.GetHorizontalAlignment();
204  VALIGNMENT valign2 = sp.GetVerticalAlignment();
205 
206  anchor.MoveBy((halign2 == HALIGN_RIGHT)?-r.W():(halign2 == HALIGN_CENTER)?-r.W()/2:0 ,
207  (valign2 == VALIGN_BOTTOM)?-r.H():(valign2 == VALIGN_CENTER)?-r.H()/2:0);
208 
209  return Rectangle(anchor.X() , anchor.Y() , newwidth , newheight);
210 }
