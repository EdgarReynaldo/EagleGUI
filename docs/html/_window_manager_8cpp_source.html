90 void EagleWindowManager::CloseWindows() {
91  if (window_count < 1) {
92  return;
93  }
94 
95  WINMAP winmap;
96 
97  EAGLE_ASSERT(manager_mutex && manager_mutex->Valid());
98 
99  ThreadLockMutex(our_thread , manager_mutex);
100  winmap = window_map;
101  ThreadUnLockMutex(our_thread , manager_mutex);
102 
103  for (WMIT it = winmap.begin() ; it != winmap.end() ; ++it) {
104  if (it->second != 0) {
105  EAGLE_ASSERT(GetValidById(it->first));
106  EagleLog() << StringPrintF("EagleWindowManager::CloseWindows - About to close window %s." , it->second->FullName()) << std::endl;
107  DestroyWindow(it->first);
108  }
109  }
110 
111  EAGLE_ASSERT(window_count == 0);
112 }

116 EagleGraphicsContext* EagleWindowManager::CreateGraphicsContext(std::string objname , int width , int height , int flags) {
117 
118  EagleGraphicsContext* window = PrivateCreateWindow(objname , width , height , flags);

145 void EagleWindowManager::DestroyWindow(int window_eid) {
146 
147  EAGLE_ASSERT(manager_mutex && manager_mutex->Valid());
148 
149 
150  ThreadLockMutex(our_thread , manager_mutex);
151 
152  WMIT it = window_map.find(window_eid);
153  if (it == window_map.end()) {
154  EagleWarn() << "EagleWindowManager::DestroyWindow - attempting to destroy an unregistered window. Ignored." << std::endl;
155  ThreadUnLockMutex(our_thread , manager_mutex);
156  return;
157  }
158  if (it->second == (EagleGraphicsContext*)0) {
159  EagleWarn() << "EagleWindowManager::DestroyWindow - attempting to destroy null window. Ignored." << std::endl;
160  ThreadUnLockMutex(our_thread , manager_mutex);
161  return;
162  }
163 
164  EagleGraphicsContext* window = window_map[window_eid];
165 
166  ThreadUnLockMutex(our_thread , manager_mutex);
167 
168  EAGLE_ASSERT(parent_system);
169 
170  EAGLE_ASSERT(window);
171 
172  parent_system->GetSystemQueue()->StopListeningTo(window);
173 
174  EagleInfo() << StringPrintF("EagleWindowManager::DestroyWindow - destroying %s" , window->FullName()) << std::endl;
175 
176  delete window;/// Must destroy window outside of lock! Or else deadlock!
177 
178 
179  ThreadLockMutex(our_thread , manager_mutex);
180 
181  window_map[window_eid] = 0;/// mark destroyed windows as null
182 
183  --window_count;
184 
185  ThreadUnLockMutex(our_thread , manager_mutex);
186 
187 }
