65 void LayoutRectangle::SetLayoutRectangle(Rectangle outer_area , Rectangle layout_area) {

91 Rectangle LayoutArea(Rectangle outer_area , LayoutRectangle layout_area) {
92  float x = outer_area.X();
93  float y = outer_area.Y();
94  float w = outer_area.W();
95  float h = outer_area.H();
96 
97  // apply percentages from layout_area
98  float nx = x + layout_area.fx*w;
99  float ny = y + layout_area.fy*h;
100  float nw = w*layout_area.fw;
101  float nh = h*layout_area.fh;
102 
103  // transform to integer boundaries, shrink to prevent overlap
104  int nxpos = (int)(ceil(nx));
105  int nypos = (int)(ceil(ny));
106  int nwid = (int)(floor(nw));
107  int nht = (int)(floor(nh));
108 
109  return Rectangle(nxpos , nypos , nwid , nht);
110 }
