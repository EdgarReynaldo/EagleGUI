38 void ScrollArea::ResetScrollbars() {
39  if (our_scroll_widget) {
40  Rectangle our_area = GetViewRectangle(false , false);
41  Rectangle their_area = our_scroll_widget->OuterArea();
42 
43 
44  hfit = their_area.W() <= our_area.W();
45  vfit = their_area.H() <= our_area.H();
46 
47  hmax = hfit?our_area.W():their_area.W();
48  hmin = hfit?their_area.W():our_area.W();
49 
50  vmax = vfit?our_area.H():their_area.H();
51  vmin = vfit?their_area.H():our_area.H();
52 
53  if (!vfit) {
54  hfit = their_area.W() <= (our_area.W() - scrollbarsize);
55  }
56  else if (!hfit) {
57  vfit = their_area.H() <= (our_area.H() - scrollbarsize);
58  }
59  if (!vfit) {
60  hmax = hfit?(our_area.W() - scrollbarsize):their_area.W();
61  hmin = hfit?their_area.W():(our_area.W() - scrollbarsize);
62  }
63  if (!hfit) {
64  vmax = vfit?(our_area.H() - scrollbarsize):their_area.H();
65  vmin = vfit?their_area.H():(our_area.H() - scrollbarsize);
66  }
67 
68  hscrollbar->SetupView(hmax , hmin);
69  vscrollbar->SetupView(vmax , vmin);
70 
71  if (Flags().FlagOn(VISIBLE)) {
72  /// Hide and disable the scrollbars
73  hscrollbar->SetVisibleState(!hfit);
74  vscrollbar->SetVisibleState(!vfit);
75  }
76  if (Flags().FlagOn(ENABLED)) {
77  /// Hide and disable the scrollbars
78  hscrollbar->SetVisibleState(!hfit);
79  vscrollbar->SetVisibleState(!vfit);
80  }
81 
82  hscrollbar->SetScrollPercent(0.0f);
83  vscrollbar->SetScrollPercent(0.0f);
84 
85  RepositionAllChildren();
86  }
87 }

174  (void)thread;

176  if (e.widget.from == hscrollbar || e.widget.from == vscrollbar) {

190 int ScrollArea::PrivateHandleEvent(EagleEvent ee) {
191  int ret = DIALOG_OKAY;
192  if (ee.type == EAGLE_EVENT_MOUSE_BUTTON_DOWN) {
193  if (ee.mouse.button == 3) {
194  Rectangle viewport = GetViewRectangle();
195  if (viewport.Contains(ee.mouse.x , ee.mouse.y)) {
196  drag = true;
197  anchorx = ee.mouse.x;
198  anchory = ee.mouse.y;
199  anchorxscroll = hscrollbar->GetScrollValue();
200  anchoryscroll = vscrollbar->GetScrollValue();
201  ret |= DIALOG_INPUT_USED;
202  }
203  }
204  }
205  if (ee.type == EAGLE_EVENT_MOUSE_AXES) {
206  if (drag) {
207  int dx = ee.mouse.x - anchorx;
208  int dy = ee.mouse.y - anchory;
209  hscrollbar->SetScroll(anchorxscroll - dx);
210  vscrollbar->SetScroll(anchoryscroll - dy);
211  ret |= DIALOG_INPUT_USED;
212  }
213  else if (ee.mouse.dz || ee.mouse.dw) {
214  hscrollbar->ScrollBy(ee.mouse.dw);
215  vscrollbar->ScrollBy(ee.mouse.dz);
216  }
217  }
218  if (ee.type == EAGLE_EVENT_MOUSE_BUTTON_UP) {
219  if (ee.mouse.button == 3) {drag = false;}
220  }
221  return ret;
222 }

299 Rectangle ScrollArea::RequestWidgetArea(int widget_slot , int newx , int newy , int newwidth , int newheight) {
300 
301  /// We control the position of all widgets
302  (void)newx;
303  (void)newy;
304 
305  /// Slot 2 is reserved for the widget to view, give it the leftover room
306  if (widget_slot == 2) {
307  return GetViewRectangle();
308  }
309 
310  /// Slots 0 and 1 are reserved for the scrollbars, and they can take up space used for the widget
311  if (widget_slot == 0 || widget_slot == 1) {
312  /// We handle the sizing and positioning of our scroll bars
313  (void)newx;
314  (void)newy;
315  (void)newwidth;
316  (void)newheight;
317  }
318 
319  /// Slot 0 is the horizontal scroll bar
320  if (widget_slot == 0) {
321  int wx = InnerArea().X();
322  int wy = InnerArea().Y();
323  int ww = InnerArea().W();
324  int wh = scrollbarsize;
325 
326  if (!hfit) {
327  ww -= scrollbarsize;
328  if (onleft) {
329  wx += scrollbarsize;
330  }
331  }
332  if (!ontop) {
333  wy = InnerArea().BRY() - scrollbarsize;
334  }
335  return Rectangle(wx,wy,ww,wh);
336  }
337  /// Slot 1 is the vertical scroll bar
338  if (widget_slot == 1) {
339  int wx = InnerArea().X();
340  int wy = InnerArea().Y();
341  int ww = scrollbarsize;
342  int wh = InnerArea().H();
343  if (!vfit) {
344  wh -= scrollbarsize;
345  if (ontop) {
346  wy += scrollbarsize;
347  }
348  }
349  if (!onleft) {
350  wx = InnerArea().BRX() - scrollbarsize;
351  }
352  return Rectangle(wx,wy,ww,wh);
353  }
354  return BADRECTANGLE;
355 }
