41 void BasicIcon::PrivateDisplay(EagleGraphicsContext* win , int xpos , int ypos) {
42  if (!icon_image) {return;}
43 
44  Rectangle src(0 , 0 , icon_image->W() , icon_image->H());
45  if (stretch_icon) {
46  Rectangle dest = InnerArea();
47  dest.MoveBy(xpos,ypos);
48  win->DrawStretchedRegion(icon_image , src , dest);
49  }
50  else {
51  int x = xpos + (InnerArea().W() - icon_image->W())/2;
52  int y = ypos + (InnerArea().H() - icon_image->H())/2;
53  win->Draw(icon_image , InnerArea().X() + x , InnerArea().Y() + y);
54  }
55 }
