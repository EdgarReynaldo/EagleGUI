180 void NPAREA::PaintOutsideGradient(EagleGraphicsContext* win , EagleColor outer , EagleColor inner) {
181  for (int i = 0 ; i < 9 ; ++i) {
182  CELL_AREA c = (CELL_AREA)i;
183 
184  if (c == CELL_AREA_MIDDLEMIDDLE) {continue;}/// Skip the middle cell
185 
186  HCELL_AREA hc = (HCELL_AREA)(i%3);
187  VCELL_AREA vc = (VCELL_AREA)(i/3);
188 
189  Rectangle r = GetNPCell(hc,vc);
190 
191  bool isleft = (hc == HCELL_LEFT);
192  bool istop = (vc == VCELL_TOP);
193 
194  /// The colors for the corners of our shaded quad
195  EagleColor cornercols[2][2];/// Top left, top right, bottom left, bottom right
196 
197  switch (c) {
198  case CELL_AREA_MIDDLELEFT :
199  case CELL_AREA_MIDDLERIGHT:
200  case CELL_AREA_TOPMIDDLE :
201  case CELL_AREA_BOTTOMMIDDLE:
202  {
203  if (hc == HCELL_LEFT) {
204  cornercols[0][0] = outer;///< top left
205  cornercols[1][0] = outer;///< bottom left
206  cornercols[0][1] = inner;///< top right
207  cornercols[1][1] = inner;///< bottom right
208  }
209  else if (hc == HCELL_CENTER) {
210  /// Top or bottom
211  if (vc == VCELL_TOP) {
212  cornercols[0][0] = outer;///< top left
213  cornercols[0][1] = outer;///< top right
214  cornercols[1][0] = inner;///< bottom left
215  cornercols[1][1] = inner;///< bottom right
216  }
217  else if (vc == VCELL_BOTTOM) {
218  cornercols[0][0] = inner;///< top left
219  cornercols[0][1] = inner;///< top right
220  cornercols[1][0] = outer;///< bottom left
221  cornercols[1][1] = outer;///< bottom right
222  }
223  }
224  else if (hc == HCELL_RIGHT) {/// right
225  cornercols[0][0] = inner;///< top left
226  cornercols[1][0] = inner;///< bottom left
227  cornercols[0][1] = outer;///< top right
228  cornercols[1][1] = outer;///< bottom right
229  }
230  }
231  break;
232  /// Paint the corners using a shaded quad, this will create beveled edges, not a rounded rectangle
233  case CELL_AREA_TOPLEFT :
234  case CELL_AREA_TOPRIGHT :
235  case CELL_AREA_BOTTOMLEFT :
236  case CELL_AREA_BOTTOMRIGHT :
237  {
238  if (isleft) {
239  if (istop) {/// top left
240  cornercols[0][0] = outer;/// top left
241  cornercols[0][1] = outer;///col;/// top right
242  cornercols[1][0] = outer;/// bottom left
243  cornercols[1][1] = inner;/// bottom right
244  }
245  else {/// bottom left
246  cornercols[0][0] = outer;/// top left
247  cornercols[0][1] = inner;/// top right
248  cornercols[1][0] = outer;/// bottom left
249  cornercols[1][1] = outer;/// bottom right
250  }
251  }
252  else {
253  if (istop) {/// top right
254  cornercols[0][0] = outer;/// top left
255  cornercols[0][1] = outer;/// top right
256  cornercols[1][0] = inner;/// bottom left
257  cornercols[1][1] = outer;/// bottom right
258  }
259  else {/// bottom right
260  cornercols[0][0] = inner;/// top left
261  cornercols[0][1] = outer;/// top right
262  cornercols[1][0] = outer;/// bottom left
263  cornercols[1][1] = outer;/// bottom right
264  }
265  }
266  }
267  break;
268  default :
269  break;
270  };
271 
272  win->DrawShadedQuad(
273  r.X() , r.Y() , cornercols[0][0] ,
274  r.X() , r.BY() , cornercols[1][0] ,
275  r.RX() , r.BY() , cornercols[1][1] ,
276  r.RX() , r.Y() , cornercols[0][1]

285  ((ypos >= pos.Y()) && (ypos <= pos.Y() + top + height + bottom))) {

307 Rectangle NPAREA::GetNPCell(HCELL_AREA hcell , VCELL_AREA vcell) const {
308  Pos2I p2 = pos;
309  const int cellwidths[4] = {0 , left , width , right};
310  const int cellheights[4] = {0 , top , height , bottom};
311  for (unsigned int j = 0 ; j <= vcell ; ++j) {
312  p2.MoveBy(0 , cellheights[j]);

628 Rectangle WIDGETAREA::CellBox(BOX_TYPE box , VCELL_AREA vcell , HCELL_AREA hcell) const {

635  &WIDGETAREA::BorderArea,
636  &WIDGETAREA::PaddingArea,
637  &WIDGETAREA::InnerArea

642  }

645  int cellx = (int)hcell;
646  Pos2I cellpos = pos;
647  const int cwidths[3][3] = {
648  {
649  margin.left ,

651  margin.right
652  },
653  {
654  border.left,
655  PaddingAreaWidth(),
656  border.right
657  },
658  {
659  padding.left,
660  InnerAreaWidth(),
661  padding.right
662  }
663  };
664  const int cheights[3][3] = {
665  {
666  margin.top ,
667  BorderAreaHeight(),
668  margin.bottom
669  },
670  {
671  border.top,
672  PaddingAreaHeight(),
673  border.bottom
674  },
675  {
676  padding.top,
677  InnerAreaHeight(),
678  padding.bottom
679  }
680  };
