110 void* Allegro5EventThreadProcess(EagleThread* thread , void* event_handler) {
111 
112  Allegro5EventHandler* a5_event_handler = dynamic_cast<Allegro5EventHandler*>((EagleEventHandler*)event_handler);
113  EAGLE_ASSERT(a5_event_handler);
114  EAGLE_ASSERT(a5_event_handler->Valid());
115 
116  ALLEGRO_EVENT_QUEUE* queue = a5_event_handler->event_queue;
117  ALLEGRO_EVENT_SOURCE* main_source = &(a5_event_handler->main_source);
118  EagleConditionVar* cond_var = a5_event_handler->cond_var;
119 
120  EagleInfo() << StringPrintF("Allegro5EventThreadProcess process starting on EagleThread %p" , thread) << std::endl;
121 
122  bool close = false;
123  while (!close) {
124  if (thread->ShouldStop()) {
125  EagleInfo() << "Allegro5EventThreadProcess : ShouldStop = TRUE" << std::endl;
126  }
127  ALLEGRO_EVENT ev;
128  al_wait_for_event(queue , &ev);
129 
130 // EagleLog() << "Event #" << ev.type << " received." << std::endl;
131  if (ev.type == EAGLE_EVENT_USER_START) {
132  if (ev.user.source == main_source) {// signalled through queue by another thread
133  switch (ev.user.data1) {
134  case CLOSE_EVENT_THREAD :
135  EagleInfo() << StringPrintF("Allegro5EventThreadProcess signaled to close thread %p." , thread) << std::endl;
136  close = true;
137  break;
138  default :
139  EagleWarn() << "Event not recognized by event thread process." << std::endl;
140  break;
141 
142  }
143  }
144  }
145  else {
146  EagleEvent ee = GetEagleEvent(ev);
147  ee.source = a5_event_handler;
148  a5_event_handler->PushEvent(ee , thread);
149  cond_var->BroadcastCondition();// alert any thread waiting on the condition (an event)
150  }
151  }
152 
153  EagleInfo() << StringPrintF("Allegro5EventThreadProcess process stopping on EagleThread %p" , thread) << std::endl;
154 
155  return event_handler;
156 }
