52 bool SendOutputToFile(const std::string& filepath , const std::string& header , bool append) {
53  EagleLogger::Instance().RemoveOutput(log_file.log_file);
54  log_file.log_file.close();
55  log_file.log_file.clear();
56  ios::openmode mode = ios::out;
57  if (append) {mode |= ios::app;}
58  log_file.log_file.open(filepath.c_str() , mode);
59  bool success = log_file.log_file.good();
60  if (success) {
61  EagleLogger::Instance().AddOutput(log_file.log_file);
62  log_file.log_file << header.c_str() << endl;
63  }
64  return success;
65 }
