48 void EagleMutex::DoLock(EagleThread* callthread , std::string callfunc) {
49 
50  EAGLE_MUTEX_STATE oldstate = state;
51  EagleThread* oldowner = owner;
52 
53  LogThreadState(callthread , callfunc.c_str() , MTX_WAITLOCK);
54 
55  if (NOT_A_THREAD != callthread) {/// don't track state, for power users only
56  if (oldstate == MTX_ISLOCKED) {
57  if (callthread == oldowner) {/// Already locked by this thread
58  EAGLE_ASSERT(Recursive());/// Must be a recursive mutex
59  }
60  }
61  }
62 
63  PrivateLock();
64 
65  LogThreadState(callthread , callfunc.c_str() , MTX_ISLOCKED);
66 }

70 bool EagleMutex::DoLockWaitFor(EagleThread* callthread , std::string callfunc , double timeout) {
71 
72  if (NOT_A_THREAD != callthread) {/// don't track state, for power users only
73  if (state == MTX_ISLOCKED) {
74  if (callthread == owner) {
75  EAGLE_ASSERT(Recursive());
76  }

203 void EagleMutex::LogThreadState(EagleThread* t , const char* func , EAGLE_MUTEX_STATE tstate) {
204  if (NOT_A_THREAD == t) {/// Free for all - Thread state not tracked
205  state = tstate;
206  if (state == MTX_ISLOCKED) {
207  owner = NOT_A_THREAD;
208  }
209  }
210  else {
211  if (t) {
212  t->SetState(tstate);
213  if (tstate == MTX_UNLOCKED) {
214  t->SetCaller("Unknown function");
215  }
216  else {
217  t->SetCaller(func);
218  }
219  }
220  state = tstate;
221  if (state == MTX_ISLOCKED) {
222  owner = t;
223  ++lock_count;
224  }
225  if (state == MTX_UNLOCKED) {
226  --lock_count;
227 
228  EAGLE_ASSERT(lock_count >= 0);
229 
230  if (!Recursive()) {
231  EAGLE_ASSERT(lock_count == 0);
232  }
233  if (lock_count == 0) {
234  owner = NOT_A_THREAD;
235  }
236  }
237  }
238 #ifdef EAGLE_DEBUG_MUTEX_LOCKS
239  if (log) {
240  std::string threadname = "Main Thread";
241  int threadid = 0;
242  if (t) {
243  if (t == NOT_A_THREAD) {
244  threadname = "NOT_A_THREAD";
245  threadid = -1;
246  }
247  else {
248  threadname = t->FullName();
249  threadid = t->ID();
250  }
251  }
252  std::string our_name = FullName();
253  ThreadLog() << StringPrintF("%50s ID %3d %s on %60s in function %-80s" ,
254  threadname.c_str() , threadid , EagleMutexStateStr(tstate) , our_name.c_str() , func) << std::endl;
255  }
256 #endif
257  return;
258 }
