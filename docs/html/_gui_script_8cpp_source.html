76 bool EGSDeclaration::ParseDeclaration(std::string declaration) {
77  Clear();
78  /// CLASSNAME OBJNAME = ATTRIBUTE:VALUE;
79  std::vector<std::string> split = SplitByDelimiterString(declaration , "=");
80  if (split.size() != 2) {
81  if (split.size() == 1) {
82  EagleError() << StringPrintF("No right hand side in EGSdeclaration \"%s\"" , declaration.c_str()) << std::endl;
83  return false;
84  }
85  else {
86  EagleError() << StringPrintF("Too many equals signs in EGSdeclaration \"%s\"" , declaration.c_str()) << std::endl;
87  return false;
88  }
89  }
90  EAGLE_ASSERT(split.size() == 2);
91  std::string lhs = split[0];
92  std::string rhs = split[1];
93 
94  std::vector<std::string> names = SplitByDelimiterString(lhs , " ");
95  if (names.size() != 2) {
96  if (names.size() == 1) {
97  EagleError() << StringPrintF("No object name in EGSdeclaration \"%s\"" , declaration.c_str()) << std::endl;
98  return false;
99  }
100  else {
101  EagleError() << StringPrintF("Too much whitespace in EGSdeclaration \"%s\"" , declaration.c_str()) << std::endl;
102  return false;
103  }
104  }
105  EAGLE_ASSERT(names.size() == 2);
106  class_name = names[0];
107  object_name = names[1];
108  attribute_value_set = rhs;
109  attribute_value_map = ParseAttributeSet(attribute_value_set);
110 
111  EAGLE_ASSERT(!attribute_value_map.Empty());
112 
113  return true;
114 }

149 bool EagleGuiScript::ReadScriptFile(std::string script_file_name) {
150  /// Read file into 'script'
151  FILE* file = fopen(script_file_name.c_str() , "r");
152  if (!file) {
153  EagleError() << StringPrintF("EagleGuiScript::ReadScriptFile : Could not open file %s for reading.\n" , script_file_name.c_str());
154  return false;
155  }
156 
157  /// Get length
158  int oldcount = 0 , count = 0;
159  while(!feof(file) && !ferror(file)) {
160  fgetc(file);
161  ++count;
162  }
163  if (ferror(file)) {
164  EagleError() << "EagleGuiScript::ReadScriptFile : error reading file size" << endl;
165  fclose(file);
166  return false;
167  }
168 
169  rewind(file);
170 
171  script_file_contents.resize(count);
172  oldcount = count;
173  count = 0;
174  while (!feof(file) && !ferror(file)) {
175  script_file_contents[count++] = fgetc(file);
176  }
177  if (ferror(file)) {
178  EagleError() << "EagleGuiScript::ReadScriptFile : error reading file" << endl;
179  fclose(file);
180  return false;
181  }
182  if (oldcount != count) {
183  EagleError() << "EagleGuiScript::ReadScriptFile : Did not fully read file" << endl;
184  fclose(file);
185  return false;
186  }
187 
188  fclose(file);
189 
190  script_file_path = script_file_name;
191 
192  return true;
193 }

199 EagleGuiScript::SECTION_MAP EagleGuiScript::ReadSections(std::string script_file) {
200 
201  SECTION_MAP smap;
202 
203  /// Read everything not in [section] tags into the current section
204  std::string section = "Global";
205  std::string content = "";
206 
207  unsigned int brace_start = script_file.find_first_of('[');
208  unsigned int brace_stop = script_file.find_first_of(brace_start , ']');
209  unsigned int content_start = brace_stop + 1;
210  unsigned int content_stop = script_file.find_first_of(content_start , '[');
211 
212  /// Global content
213  content = script_file.substr(0 , brace_start);
214  smap[section] = content;
215 
216  /// Named sections
217  while (brace_start != std::string::npos && brace_stop != std::string::npos) {
218  section = script_file.substr(brace_start + 1 , brace_stop);
219  content = script_file.substr(content_start , content_stop);
220  smap[section] = content;
221  brace_start = content_stop;
222  brace_stop = script_file.find_first_of(brace_start , ']');
223  content_start = brace_stop + 1;
224  content_stop = script_file.find_first_of(content_start , '[');
225  }
226 
227  return section_map;
228 }

232 std::vector<std::string> EagleGuiScript::GetSectionLines(std::string content) {
233  std::vector<std::string> lines = SplitByNewLinesChomp(content);
234  std::vector<std::string> adjusted;
235  string current = lines[0];
236 
237 
238  /// Assume there is a statement. If we get an empty line, we push it back. If there are more empty lines after that we skip them.
239 
240  int i = 0;
241  do {
242  /// Skip empty lines
243  while (lines[i].compare("") == 0 && i < (int)lines.size()) {
244  ++i;
245  }
246  /// We now have content, or EOF
247  if (i == (int)lines.size()) {
248  break;
249  }
250  current = "";
251  while (lines[i].compare("") != 0 && i < (int)lines.size()) {
252  if (lines[i][0] != '#') {
253  current.append(lines[i]);
254  current.append(";");
255  }
256  ++i;
257  }
258  adjusted.push_back(current);
259 
260  } while (i < (int)lines.size());
261 
262  return adjusted;
263 }
