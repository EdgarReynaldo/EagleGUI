46 void BasicText::RefreshTextPosition() {
47 
48  lines = SplitByNewLinesNoChomp(text);
49  nlines = lines.size();
50  if (nlines < 1) {return;}
51 
52  EAGLE_ASSERT(text_font);
53  EAGLE_ASSERT(text_font->Valid());
54 
55  int lineheight = fontheight + linespacing;
56  maxwidth = 0;
57  totalheight = fontheight*nlines + linespacing*(nlines-1);
58 
59  lineareas.clear();
60 
61 
62  int y = InnerArea().Y() + vpadding;
63  if (valign == VALIGN_CENTER) {
64  y = InnerArea().CY() - totalheight/2;
65  }
66  if (valign == VALIGN_BOTTOM) {
67  y = InnerArea().BRY() - vpadding - totalheight;
68  }
69  texty = y;
70  for (int i = 0 ; i < nlines ; ++i) {
71  int w = text_font->Width(lines[i].c_str());
72  int x = InnerArea().X() + hpadding;
73  if (halign == HALIGN_CENTER) {
74  x = InnerArea().CX() - w/2;
75  }
76  if (halign == HALIGN_RIGHT) {
77  x = InnerArea().BRX() - hpadding - w;
78  }
79  if (w > maxwidth) {
80  maxwidth = w;
81  textx = x;
82  }
83  lineareas.push_back(Rectangle(x,y,w,fontheight));
84  y += lineheight;
85  }
86  SetBgRedrawFlag();
87 }

211 void BasicText::DrawText(EagleGraphicsContext* win , int xpos , int ypos , EagleColor c) {
212  EAGLE_ASSERT(win);
213  EAGLE_ASSERT(win->Valid());
214 
215  if (nlines < 1) {return;}
216 
217  EAGLE_ASSERT(text_font && text_font->Valid());
218 
219  Transformer* transformer = win->GetTransformer();
220 
221  if (scale_to_fit) {
222 
223  scaling_transform = transformer->GetIdentityTransform();
224 
225  scaling_transform.Translate(-textx , -texty);
226  scaling_transform.Scale((float)InnerArea().W()/maxwidth , (float)InnerArea().H()/totalheight);
227  scaling_transform.Translate(InnerArea().X() , InnerArea().Y());
228 
229  scaling_transform *= transformer->GetViewTransform();
230 
231  transformer->PushViewTransform(scaling_transform);
232  }
233 
234 
235  win->HoldDrawing();
236  for (int i = 0 ; i < nlines ; ++i) {
237  string s = GetGuiText(lines[i]);
238  Rectangle r = lineareas[i];
239  win->DrawTextString(text_font , s , r.X() + xpos , r.Y() + ypos , c , HALIGN_LEFT , VALIGN_TOP);
240  }
241  win->ReleaseDrawing();
242 
243  if (scale_to_fit) {
244  win->GetTransformer()->PopViewTransform();
245  }
246 /// win->DrawMultiLineTextString(text_font , text , tx + xpos , ty + ypos , c , linespacing , halign , valign);
247 }
