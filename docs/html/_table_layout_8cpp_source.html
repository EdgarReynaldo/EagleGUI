95 Rectangle TableBodyLayout::WidgetRectangle(unsigned int columnx , unsigned int rowy) {
96  Rectangle rcell = CellRectangle(columnx , rowy);
97 
98  WidgetBase* widget = GetWidget(rowy*ncols + columnx);
99 
100  if (!widget) {
101  return BADRECTANGLE;
102  }
103 
104  const TableCell* tcell = &cell_map[widget];
105 
106  if (tcell->align) {
107  HALIGNMENT halignc = tcell->halign;
108  VALIGNMENT valignc = tcell->valign;
109  int wx = rcell.X();
110  int wy = rcell.Y();
111  int ww = widget->OuterArea().W();
112  int wh = widget->OuterArea().H();
113  if (halignc == HALIGN_CENTER) {
114  wx += (rcell.W() - ww)/2;
115  }
116  else if (halignc == HALIGN_RIGHT) {
117  wx += rcell.W() - ww;
118  }
119  if (valignc == VALIGN_CENTER) {
120  wy += (rcell.H() - wh)/2;
121  }
122  else if (valignc == VALIGN_BOTTOM) {
123  wy += rcell.H() - wh;
124  }
125  Rectangle wrect(wx,wy,ww,wh);
126  return wrect;
127  }
128  else {
129  /// Fill cell span
130  return rcell;
131  }
132  return BADRECTANGLE;/// Not reached
133 }

137 Rectangle TableBodyLayout::CellRectangle(unsigned int columnx , unsigned int rowy) {
138  if (columnx >= colwidths.size()) {
139  return BADRECTANGLE;
140  }
141  if (rowy >= rowheights.size()) {
142  return BADRECTANGLE;
143  }
144  WidgetBase* widget = GetWidget(rowy*ncols + columnx);
145 
146  if (widget) {
147  const TableCell* cell = &cell_map[widget];
148  unsigned int spanwidth = 0;
149  unsigned int spanheight = 0;
150  unsigned int xpos = InnerArea().X();
151  unsigned int ypos = InnerArea().Y();
152 
153  for (unsigned int col = 1 ; col < cell->startcol ; ++col) {
154  xpos += colwidths[col];
155  }
156  for (unsigned int row = 0 ; row < cell->startrow ; ++row) {
157  ypos += rowheights[row];
158  }
159 
160  for (unsigned int col = 0 ; col < cell->colspan ; ++col) {
161  spanwidth += colwidths[col + cell->startcol];
162  }
163  for (unsigned int row = 0 ; row < cell->rowspan ; ++row) {
164  spanheight += rowheights[row + cell->startrow];
165  }
166 
167  Rectangle r(xpos , ypos , spanwidth , spanheight);
168 
169  return r;
170  }
171  return SlotRectangle(columnx , rowy);
172 }

198 void TableBodyLayout::PlaceWidgetInTable(WidgetBase* w , int slot , TableCell cell) {
199  if (slot >= (int)wchildren.size()) {
200  throw EagleException(StringPrintF("TableBodyLayout::PlaceWidgetInTable() : slot (%d) out of range! (%u)\n" , slot , wchildren.size()));
201  }
202 
203  if (cell.Area() == 0) {
204  throw EagleException("TableBodyLayoyut::PlaceWidgetInTable() : cell area is zero!\n");
205  }
206 
207  Rectangle rtable(0,0,ncols,nrows);
208  Rectangle rcell(cell.startcol , cell.startrow , cell.colspan , cell.rowspan);
209  if (!rtable.Contains(rcell)) {
210  throw EagleException("TableBodyLayout::PlaceWidgetInTable() : cell doesn't fit the table!\n");
211  }
212 
213  WidgetBase* old = GetWidget(slot);
214  if (old) {
215  TableCell tcell = cell_map[old];
216  for (unsigned int r = tcell.startrow ; r < tcell.startrow + tcell.rowspan ; ++r) {
217  for (unsigned int c = tcell.startcol ; c < tcell.startcol + tcell.colspan ; ++c) {
218  cell_table[r][c] = 0;
219  }
220  }
221  cell_map.erase(cell_map.find(old));
222  }
223  if (w) {
224  TableCell tcell = cell_map[w];
225  for (unsigned int r = tcell.startrow ; r < tcell.startrow + tcell.rowspan ; ++r) {
226  for (unsigned int c = tcell.startcol ; c < tcell.startcol + tcell.colspan ; ++c) {
227  cell_table[r][c] = w;
228  }
229  }
230  cell_map[w] = cell;
231  }
232 
233  LayoutBase::PlaceWidget(w,slot);
234 }

238 TableCell TableBodyLayout::NextFreeCell() {
239 
240  /// Start with blank table
241  TableVec2D<TableCell> celltable(nrows , ncols , TableCell());
242  for (unsigned int i = 0 ; i < nrows ; ++i) {
243  for (unsigned int j = 0 ; j < ncols ; ++j) {
244  celltable[i][j].Place(i,j).Size(0,0);
245  }
246  }
247 
248  /// Remove used cells
249  for (std::map<WidgetBase* , TableCell>::iterator it = cell_map.begin() ; it != cell_map.end() ; ++it) {
250  const TableCell& cell = it->second;
251  for (unsigned int i = cell.startrow ; i < cell.startrow + cell.rowspan ; ++i) {
252  for (unsigned int j = cell.startcol ; j < cell.startcol + cell.colspan ; ++j) {
253  celltable[i][j].Size(1,1);
254  }
255  }
256  }
257 
258  /// Find first unused cell
259  for (unsigned int i = 0 ; i < celltable.NRows() ; ++i) {
260  for (unsigned int j = 0 ; j < celltable.NCols() ; ++j) {
261  if (celltable[i][j].Area() == 0) {
262  return celltable[i][j];
263  }
264  }
265  }
266  return TableCell();
267 }

283 Rectangle TableBodyLayout::RequestWidgetArea(int widget_slot , int newx , int newy , int newwidth , int newheight) {
284  (void)newx;
285  (void)newy;
286  (void)newwidth;
287  (void)newheight;
288  WidgetBase* widget = GetWidget(widget_slot);
289  if (!widget) {
290  return BADRECTANGLE;
291  }
292  Rectangle wrect = WidgetRectangle(widget_slot%ncols , widget_slot/ncols);
293  if (wrect != BADRECTANGLE) {
294  if (newwidth != INT_MAX || newheight != INT_MAX) {
295  ///< Adjust the widget rectangle for the new width and height, newx and newy are ignored
296 
297  const TableCell* tcell = &cell_map[widget];
298 
299  int ox = 0;
300  int oy = 0;
301  int dw = 0;
302  int dh = 0;
303 
304  if (newwidth == INT_MAX) {
305  dw = newwidth - widget->OuterArea().W();
306  newwidth = widget->OuterArea().W();
307  }
308  if (newheight == INT_MAX) {
309  dh = newheight - widget->OuterArea().H();
310  newheight = widget->OuterArea().H();
311  }
312 
313  if (tcell->align) {
314  if (tcell->halign == HALIGN_CENTER) {
315  /// We need to center the widget based on the new width
316  ox += -dw/2;
317  }
318  else if (tcell->halign == HALIGN_RIGHT) {
319  ox += -dw;
320  }
321  if (tcell->valign == VALIGN_CENTER) {
322  /// We need to center the widget based on the new width
323  oy += -dh/2;
324  }
325  else if (tcell->valign == VALIGN_BOTTOM) {
326  oy += -dh;
327  }
328  wrect.MoveBy(ox,oy);
329  wrect.SetDimensions(newwidth , newheight);
330  }
331  }
332  }
333  return wrect;
334 }
