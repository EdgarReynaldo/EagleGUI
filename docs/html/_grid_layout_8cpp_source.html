39 Rectangle GridLayout::RequestWidgetRectangle(int slot_index) const {
40 
41  const WidgetBase* widget = GetWidget(slot_index);
42 
43  if (!widget) {
44  return BADRECTANGLE;
45  }
46 
47  int col = slot_index % ncols;
48  int row = slot_index / ncols;
49 
50  int basex = InnerArea().X() + col*(colwidth + colhspace);
51  int basey = InnerArea().Y() + row*(rowheight + rowvspace);
52 
53  int wwidth = widget->OuterArea().W();
54  int wheight = widget->OuterArea().H();
55 
56  switch (options) {
57  case GRID_ALIGNMENT_ONLY :
58  {
59  switch (halign) {
60  case HALIGN_LEFT :
61  basex += cellhpad;
62  break;
63  case HALIGN_CENTER :
64  basex += (colwidth - wwidth)/2;
65  break;
66  case HALIGN_RIGHT :
67  basex += (colwidth - wwidth) - cellhpad;
68  break;
69  default : EAGLE_ASSERT(0);break;
70  }
71  switch (valign) {
72  case VALIGN_TOP :
73  basey += cellvpad;
74  break;
75  case VALIGN_CENTER :
76  basey += (rowheight - wheight)/2;
77  break;
78  case VALIGN_BOTTOM :
79  basey += (rowheight - wheight) - cellvpad;
80  break;
81  default : EAGLE_ASSERT(0);break;
82  }
83  }; break;
84 
85  case GRID_FILL_CELL :
86  basex += cellhpad;
87  basey += cellvpad;
88 
89  wwidth = colwidth - 2*cellhpad;
90  if (wwidth < widget->AbsMinWidth()) {wwidth = widget->AbsMinWidth();}
91  wheight = rowheight - 2*cellvpad;
92  if (wheight < widget->AbsMinHeight()) {wheight = widget->AbsMinHeight();}
93  break;
94  default : EAGLE_ASSERT(0);break;
95  }
96 
97 
98  return Rectangle(basex,basey,wwidth,wheight);
99 }

182 /// Resize grid will keep all widgets that overlap the new grid, but all others will be removed
183 void GridLayout::ResizeGrid(int newcolumns , int newrows) {
184  if (newcolumns < 0) {newcolumns = 0;}
185  if (newrows < 0) {newrows = 0;}
186 
187  const int newsize = newcolumns*newrows;
188 
189  if (newsize == 0) {
190  ClearWidgets();
191  nrows = 0;
192  ncols = 0;
193  size = 0;
194  }
195  else {
196  std::vector<WidgetBase*> keep_widgets((unsigned int)newsize , (WidgetBase*)0);
197  std::vector<WidgetBase*> removed_widgets;
198  int oldindex = 0;
199  int newindex = 0;
200  const int oldrows = nrows;
201  const int oldcols = ncols;
202  int row = 0;
203  int col = 0;
204  for (row = 0 ; row < oldrows ; ++row) {
205  /// If the current row is outside the new grid, remove the widgets in that row
206  if (row >= newrows) {
207  for (col = 0 ; col < oldcols ; ++col) {
208  oldindex = row*oldcols + col;
209  RemoveWidgetFromLayout(wchildren[oldindex]);/// Places NULL in its slot
210  }
211  }
212  else {
213  for (col = 0 ; col < oldcols ; ++col) {
214  oldindex = row*oldcols + col;
215  /// If the current column is outside the new grid, remove the widget in the current row and column
216  if (col >= newcolumns) {
217  RemoveWidgetFromLayout(wchildren[oldindex]);/// Places NULL in its slot
218  }
219  else {
220  newindex = row*newcolumns + col;
221  /// Keep this widget
222  keep_widgets[newindex] = wchildren[oldindex];
223  }
224  }
225  }
226  }
227  wchildren = keep_widgets;
228  ncols = newcolumns;
229  nrows = newrows;
230  size = newsize;
231  CalculateGrid();
232  RepositionAllChildren();
233  }
234 }
