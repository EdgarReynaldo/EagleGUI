22 void* Allegro5TimerProcess(EagleThread* ethread , void* etimer) {
23  EagleTimer* eagle_timer = (EagleTimer*)etimer;
24  Allegro5Timer* eagle_a5_timer = dynamic_cast<Allegro5Timer*>(eagle_timer);
25  EAGLE_ASSERT(eagle_a5_timer);
26 
27  ALLEGRO_TIMER* timer = eagle_a5_timer->AllegroTimer();
28  ALLEGRO_EVENT_QUEUE* queue = eagle_a5_timer->AllegroEventQueue();
29  ALLEGRO_EVENT_SOURCE* event_source = eagle_a5_timer->AllegroEventSource();
30 
31  EAGLE_ASSERT(timer);
32  EAGLE_ASSERT(queue);
33  EAGLE_ASSERT(event_source);
34 
35  int counter = 0;
36  bool close = false;
37  while (!ethread->ShouldStop() && !close) {
38  ALLEGRO_EVENT ev;
39  al_wait_for_event(queue , &ev);
40 // EagleInfo() << "Event " << ev.type << std::endl;
41  if (ev.type == ALLEGRO_EVENT_TIMER && ev.timer.source == (ALLEGRO_TIMER*)eagle_a5_timer->Source()) {
42 // EagleInfo() << "TIMER_EVENT received" << std::endl;
43  ++counter;
44  eagle_a5_timer->Tick(ev.any.timestamp , ethread);///al_get_time());
45  }
46  else if (ev.type == EAGLE_EVENT_USER_START) {
47  switch (ev.user.data1) {
48  case EAGLE_MESSAGE_CREATE_TIMER :
49 // EagleInfo() << "EAGLE_MESSAGE_CREATE_TIMER received." << std::endl;
50  break;
51  case EAGLE_MESSAGE_START_TIMER :
52  al_start_timer(timer);
53 // EagleInfo() << "EAGLE_MESSAGE_START_TIMER received." << std::endl;
54  break;
55  case EAGLE_MESSAGE_STOP_TIMER :
56  al_stop_timer(timer);
57 // EagleInfo() << "EAGLE_MESSAGE_STOP_TIMER received." << std::endl;
58  break;
59  case EAGLE_MESSAGE_CLOSE_TIMER :
60  close = true;
61 // EagleInfo() << "EAGLE_MESSAGE_CLOSE_TIMER received." << std::endl;
62  break;
63  default : EAGLE_ASSERT(0);
64  break;
65  }
66  al_emit_user_event(event_source , &ev , 0);// bounce message back to synchronize states
67  }
68  }
69  return (void*)(long)counter;
70 }
