49 Rectangle RelativeLayout::RequestWidgetArea(int widget_slot , int newx , int newy , int newwidth , int newheight) {
50 
51  const WidgetBase* widget = GetWidget(widget_slot);
52  Rectangle r = widget->OuterArea();
53 
54  if (!widget) {
55  return BADRECTANGLE;
56  }
57 
58  if (newx == INT_MAX && newy == INT_MAX && newwidth == INT_MAX && newheight == INT_MAX) {
59  /// Just return the stored layout position
60  return LayoutArea(InnerArea() , layout_rectangles[widget_slot]);
61  }
62 
63  if (newx == INT_MAX) {newx = r.X();}
64  if (newy == INT_MAX) {newy = r.Y();}
65  if (newwidth == INT_MAX) {newwidth = r.W();}
66  if (newheight == INT_MAX) {newheight = r.H();}
67 
68  AdjustWidgetArea(widget , &newx , &newy , &newwidth , &newheight);
69 
70  Rectangle cl = InnerArea();
71  if (newx < cl.X()) {newx = cl.X();}/// Fit left
72  if (newy < cl.Y()) {newy = cl.Y();}/// Fit top
73  if (newx > cl.BRX()) {newx = cl.BRX();}/// Fit right
74  if (newy > cl.BRY()) {newy = cl.BRY();}/// Fit bottom
75  if (newx + newwidth > cl.BRX()) {newwidth = cl.BRX() - newx + 1;}/// Adjust width
76  if (newy + newheight > cl.BRY()) {newheight = cl.BRY() - newy + 1;}/// Adjust height
77 
78  Rectangle clipped(newx,newy,newwidth,newheight);
79 
80  EAGLE_ASSERT(clipped.Area());/// Probably don't want to set an area with zero height or width
81 
82  LayoutRectangle lr(warea.InnerArea() , clipped);
83  layout_rectangles[widget_slot] = lr;
84  return LayoutArea(InnerArea() , lr);
85 }
