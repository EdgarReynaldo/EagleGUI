51 WIDGETCOLOR WidgetColorFromName(std::string name) {
52  static const std::string names[EAGLE_NUMCOLORS] = {
53  "SDCOL",
54  "BGCOL",
55  "MGCOL",
56  "FGCOL",
57  "HLCOL",
58  "TXTCOL",
59  "FCSCOL",
60  "HVRCOL",
61  "PADCOL",
62  "BORDCOL",
63  "MARGCOL"
64  };
65  for (int i = 0 ; i < EAGLE_NUMCOLORS ; ++i) {
66  if (name.compare(names[i]) == 0) {
67  return (WIDGETCOLOR)i;
68  }
69  }
70  throw EagleException(StringPrintF("Color '%s' is not a recognized WIDGETCOLOR.\n" , name.c_str()));
71  return SDCOL;
72 }
