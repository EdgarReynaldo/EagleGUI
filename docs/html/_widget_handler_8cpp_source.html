456 void WidgetHandler::CheckRedraw() {
457  /// Check from back to front
458  typedef std::set<unsigned int> WSET;
459  typedef list<Rectangle> RLIST;
460 
461  /// Prime the list of widgets to check
462  WSET recheck;
463  for (unsigned int i = 0 ; i < drawlist.size() ; ++i) {
464  if (drawlist[i]->Flags().FlagOn(NEEDS_REDRAW)) {
465  recheck.insert(i);
466  }
467  }
468 
469  /// Prime the dirty background list - it may be altered by CheckRedraw or if a widget sets the BG_REDRAW flag
470  RLIST dirty = dbg_list;
471  dbg_list.clear();
472 
473  /// For storing new areas that get marked as dirty by widget checks
474  RLIST new_dbg_list;
475 
476  /// Run each list at least once
477  do {
478  WSET check = recheck;
479  recheck.clear();
480 
481  /// Each widget that overlaps a dirty background area needs to be redrawn
482  for (RLIST::iterator it = dirty.begin() ; it != dirty.end() ; ++it) {
483  Rectangle r = *it;
484  for (UINT i = 0 ; i < drawlist.size() ; ++i) {
485  WidgetBase* w = drawlist[i];
486  Rectangle area = w->OuterArea();
487  if (r.Overlaps(area)) {
488  if (w->Flags().FlagOff(NEEDS_REDRAW)) {
489  w->SetRedrawFlag();/// This may alter dbg_list
490  recheck.insert(i);
491  }
492  }
493  }
494  }
495 
496  /// Each widget that needs redraw needs to check for causing other widgets to redraw
497  for (WSET::iterator it = check.begin() ; it != check.end() ; ++it) {
498  WSET newcheck = CheckRedraw(*it);/// This may alter dbg_list
499  recheck.insert(newcheck.begin() , newcheck.end());
500  }
501 
502  /// Prepare the next set of dirty backgrounds
503  new_dbg_list.insert(new_dbg_list.end() , dirty.begin() , dirty.end());
504  dirty = dbg_list;
505  dbg_list.clear();
506 
507  } while (recheck.size() || dirty.size());
508 
509  dbg_list = new_dbg_list;
510 }

590 WidgetBase* WidgetHandler::NextFocus() {
591  /// focus index holds the index of the current widget with focus in the wlist
592  /// return the next widget that accepts focus if there is one in the list
593  /// can't skip to the next one if it is a WidgetHandler - need to use NextFocus() instead
594  /// skip the WidgetHandler if it returns 0 from NextFocus()
595  /// NextFocus() should return 0 when it reaches the end of it's wlist
596  int start_index = focus_index;
597  if (focus_start) {
598  start_index = -1;
599  focus_start = false;
600  } else if (wlist.size()) {
601  WidgetBase* w = wlist[focus_index];
602  WidgetHandler* wh = w->GetGui();
603  if (wh) {
604  WidgetBase* next = wh->NextFocus();
605  if (next) {return next;}
606  }
607  }
608  for (int index = start_index + 1 ; index < (int)wlist.size() ; ++index) {
609  WidgetBase* widget = wlist[index];
610  UINT flags = widget->Flags();
611  if ((flags & ENABLED) && (flags & VISIBLE) && InView(widget)) {
612  if (widget->HasGui()) {
613  WidgetHandler* handler = widget->GetGui();
614  handler->FocusStart();
615  WidgetBase* next = handler->NextFocus();
616  if (next) {
617  return next;
618  }
619  } else {
620  if (widget->AcceptsFocus()) {
621  return widget;
622  }
623  }
624  }
625  }
626  // if next focus reaches the end of the wlist it returns 0 and starts over next time
627  focus_start = true;
628  return (WidgetBase*)0;
629 }
