41 void BasicScrollButton::ResetTriangle() {
42  Triangle* t = &our_click_area;
43  const Rectangle r = InnerArea();
44  int x1 = 0;
45  int y1 = 0;
46  int x2 = 0;
47  int y2 = 0;
48  int x3 = 0;
49  int y3 = 0;
50  if (scroll_horizontal) {
51  /// Left right scroller
52  x1 = 0;
53  y1 = r.H()/2;
54  x2 = r.W();
55  x3 = x2;
56  y2 = 0;
57  y3 = r.H();
58  if (scroll_up_or_left) {
59  /// Scroll button points left
60  /// Already set up for triangle facing left
61  }
62  else {
63  /// Scroll button points right
64  /// Flip x coords
65  x1 = r.W();
66  x2 = r.W() - x2;
67  x3 = x2;
68  }
69  }
70  else {
71  /// Up down scroller
72  x1 = r.W()/2;
73  y1 = 0;
74  x2 = r.W();
75  x3 = 0;
76  y2 = r.H();
77  y3 = y2;
78  if (scroll_up_or_left) {
79  /// Scroll button points up
80  /// Already setup for triangle facing up.
81  }
82  else {
83  /// Scroll button points down
84  /// Flip y coords
85  y1 = r.H();
86  y2 = r.H() - y2;
87  y3 = y2;
88  }
89  }
90  t->SetPos(x1,y1,x2,y2,x3,y3);
91 }

112  if (scroll_button == &our_basic_button) {
113  Triangle t = our_click_area;
114  t.MoveBy(InnerArea().X() + xpos , InnerArea().Y() + ypos);
115  WidgetColorset wc = WidgetColors();

138  scroll_button->Display(win,xpos,ypos);

152 void BasicScrollButton::QueueUserMessage(const WidgetMsg& wmsg) {
153 
154  const WidgetMsg clickmessage(scroll_button , TOPIC_BUTTON_WIDGET , BUTTON_CLICKED);
155  const WidgetMsg heldmessage(scroll_button , TOPIC_BUTTON_WIDGET , BUTTON_HELD);
156 
157  const WidgetMsg releasemessage(scroll_button , TOPIC_BUTTON_WIDGET , BUTTON_RELEASED);
158 
159  if ((wmsg == clickmessage) || (wmsg == heldmessage)) {
160  Scroll();
161  }
162 
163  if (wmsg == clickmessage) {
164  RaiseEvent(WidgetMsg(this , TOPIC_BUTTON_WIDGET , BUTTON_CLICKED));
165  }
166  else if (wmsg == heldmessage) {
167  RaiseEvent(WidgetMsg(this , TOPIC_BUTTON_WIDGET , BUTTON_HELD));
168  }
169  else if (wmsg == releasemessage) {
170  RaiseEvent(WidgetMsg(this , TOPIC_BUTTON_WIDGET , BUTTON_RELEASED));
171  }
172  WidgetBase::QueueUserMessage(wmsg);
173 }
