28 std::string WidgetFlagString(unsigned int flags) {
29  static const char* fstrs[NUM_WIDGET_FLAGS] = {
30  "FLAGS_NONE",
31  "ENABLED",
32  "VISIBLE",
33  "HOVER",
34  "HASFOCUS",
35  "MOVEABLE",
36  "RESIZEABLE",
37  "NEEDS_REDRAW",
38  "NEEDS_BG_REDRAW",
39  "ALLOW_CLOSE"
40  };
41  if (!flags) {
42  return fstrs[0];
43  }
44  std::stringstream strm;
45  strm.str("");
46  bool first = true;
47  for (int i = 1 ; i < NUM_WIDGET_FLAGS ; ++i) {
48  unsigned int f = 1 << (i-1);
49  if (flags & f) {
50  if (!first) {
51  strm << " | ";
52  }
53  strm << fstrs[i];
54  first = false;
55  }
56 
57  }
58  return strm.str();
59 }

114 WidgetFlags& WidgetFlags::AddFlags(unsigned int flags_to_add) {
115  flags_to_add &= ~added_flags;/// Only add flags that are not added already
116  removed_flags &= ~flags_to_add;
117  added_flags |= flags_to_add;
118  changed_flags = removed_flags | added_flags;
