182 void EagleObjectRegistry::Register(EagleObject* object , std::string objclass , std::string name , EAGLE_ID eid /* = EAGLE_ID_UNASSIGNED*/) {
183  EAGLE_ASSERT(object);
184 
185  int id_index = -1;
186 
187  if (eid != EAGLE_ID_INVALID) {
188  /// In this case all we are doing is assigning a new name to the object
189 
190  LockOurMutex();
191 
192  CheckIdRange(eid);
193 
194  /// Remove the correct entry from the name map before inserting the new name element
195  RemoveNameEntry(eid);
196 
197  NMVALUETYPE nmval(name , eid);
198  pnamemap->insert(nmval);
199 
200  id_index = eid - start_id;
201 
202  EAGLE_ASSERT((*pinfo)[id_index].GetObject() == object);/// Make sure the object address matches for this eid
203 
204  (*pinfo)[id_index].SetShortName(name);
205  (*pinfo)[id_index].SetClassName(objclass);
206 
207  object->our_info = (*pinfo)[id_index];
208 
209  UnLockOurMutex();
210 
211  return;
212  }
213 
214  LockOurMutex();
215 
216  EAGLE_ID new_id = NextId();
217 
218  object->SetId(new_id);
219 
220 
221  if (object->log_registration) {
222  EaglePrefix("EAGLE_CREATE : ") << StringPrintF("Creating %s object '%s' at %p with eid %d\n" , objclass.c_str() , name.c_str() , object , new_id) << std::endl;
223  }
224 
225  stop_id = new_id + 1;
226 
227  id_index = new_id - start_id;
228 
229  EAGLE_ASSERT((int)pinfo->size() == id_index);
230 
231  EagleObjectInfo eoi(object , objclass , name);
232 
233  pinfo->push_back(eoi);
234 
235  object->our_info = eoi;
236 
237  (*paddressmap)[object] = new_id;
238 
239  NMVALUETYPE nmval(name , new_id);
240 
241  pnamemap->insert(nmval);
242 
243  UnLockOurMutex();
244 }

248 void EagleObjectRegistry::Unregister(EAGLE_ID eid) {
249  EAGLE_ASSERT(eid != EAGLE_ID_INVALID);
250 
251  LockOurMutex();
252 
253  CheckIdRange(eid);
254 
255  RemoveNameEntry(eid);
256 
257  int id_index = eid - start_id;
258 
259  EagleObjectInfo& eoi = (*pinfo)[id_index];
260 
261  if (!eoi.IsDestroyed() && eoi.GetObject()->log_registration) {
262  EaglePrefix("EAGLE_DESTROY : ") << StringPrintF("Destroying eagle object %s" , eoi.FullName()) << std::endl;
263  }
264 
265  if (eoi.IsDestroyed()) {
266  EagleError() << StringPrintF("Double destruction of eagle object '%s' detected." , eoi.FullName()) << std::endl;
267  EAGLE_ASSERT(!eoi.IsDestroyed());/// Throw
268  }
269  else {
270  ADDRESSMAP::iterator mit = paddressmap->find(eoi.GetObject());
271  EAGLE_ASSERT(mit != paddressmap->end());
272  paddressmap->erase(mit);
273  eoi.SetDestroyedFlag();
274  ++destruct_count;
275  }
276 
277  UnLockOurMutex();
278 
279 }
