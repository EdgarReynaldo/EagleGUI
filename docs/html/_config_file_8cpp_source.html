42 void ConfigLine::ParseLine() {
43  spacer = !line.size();
44  comment = false;
45  key = "";
46  value = "";
47  if (!spacer) {
48  if (line[0] == '#') {
49  comment = true;
50  }
51  }
52  if (!comment && !spacer) {
53  unsigned int j = line.find_first_of('=');
54  EAGLE_ASSERT(line.find_first_of('=') != std::string::npos);
55  if (j != std::string::npos) {
56  /// Should have a key value pair
57  key = line.substr(0 , j);
58  value = line.substr(j + 1);
59  while (j > 0 && isspace(key[j-1])) {--j;}
60  key = key.substr(0 , j);/// Trim trailing white space off of key
61  j = 0;
62  while (j < value.size() && isspace(value[j])) {++j;}/// Trim leading whitespace off of value
63  value = value.substr(j);
64  }
65  }
66 }

332 bool ConfigFile::LoadFromFile(const char* path) {
333 
334  Clear();
335 
336  EagleSystem* sys = Eagle::EagleLibrary::System("Any");
337 
338  EAGLE_ASSERT(sys);
339 
340  FSInfo finfo = sys->GetFileSystem()->GetFileInfo(std::string(path));
341 
342  std::string fpath = finfo.Path();
343 
344  if (!finfo.Exists()) {
345  EagleWarn() << StringPrintF("ConfigFile::LoadFromFile - failed to load file on path '%s'\n" , fpath.c_str());
346  return false;
347  }
348  if (!finfo.Mode().IsFile()) {
349  EagleError() << StringPrintF("ConfigFile::LoadFromFile - path '%s' does not refer to a file!\n" , fpath.c_str());
350  return false;
351  }
352 
353  MemFile mem(finfo);
354  if (!mem.ReadFileIntoMemory()) {
355  return false;
356  }
357 
358  contents.insert(contents.begin() , mem.Begin() , mem.End());
359 
360  mem.Clear();
361 
362  std::vector<std::string> lines = SplitByNewLinesChomp(contents);
363 
364  ConfigSection* section = &sectionmap["GLOBAL"];
365  for (int i = 0 ; i < (int)lines.size() ; ++i) {
366  std::string l = lines[i];
367 
368  unsigned int idx1 = l.find_first_of('[');
369  unsigned int idx2 = l.find_first_of(']');
370  if (idx1 != std::string::npos && idx2 != std::string::npos && idx1 < idx2 && idx1 == 0) {
371  /// Found a section name
372  std::string section_str = l.substr(idx1 + 1 , idx2 - idx1 - 1);
373  section = &sectionmap[section_str];
374  }
375  else {
376  section->AddConfigLine(l);
377  }
378  }
379  return true;
380 }
