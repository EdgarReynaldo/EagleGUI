689 EagleEvent EagleEventHandler::WaitForEvent(double timeout , EagleThread* thread) {
690  EAGLE_ASSERT(Valid());
691  EAGLE_ASSERT(HasSources());
692 
693  EagleEvent e;
694 
695  LockOurMutex(thread);
696 
697  if (!queue.empty()) {
698  // eagle event in queue

700  queue.pop_front();
701  UnLockOurMutex(thread);

706
