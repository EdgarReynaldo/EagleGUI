128 bool WidgetFactory::FreeWidget(EAGLE_ID eid) {
129 
130  WIDGET_GROUP_MAP::iterator default_group_it = widget_group_map.find(DEFAULT_GROUP_NAME);
131  EAGLE_ASSERT(default_group_it != widget_group_map.end());
132 
133  WIDGET_FREED_MAP& default_group = default_group_it->second;
134 
135  WIDGET_FREED_MAP::iterator freed_map_it = default_group.find(eid);
136  if (freed_map_it != default_group.end()) {
137  /// Found eid in default group
138  return FreeWidget(eid , default_group);
139  }
140  else {
141  /// Need to iterate over groups and search for eid
142  WIDGET_GROUP_MAP::iterator group_it = widget_group_map.begin();
143  while (group_it != widget_group_map.end()) {
144  if (group_it == default_group_it) {++group_it;continue;}
145  WIDGET_FREED_MAP& freed_map = group_it->second;
146  freed_map_it = freed_map.find(eid);
147  if (freed_map_it != freed_map.end()) {
148  /// Found eid in freed_map
149  return FreeWidget(eid , freed_map);
150  }
151  ++group_it;
152  }
153  }
154  return false;
155 }

169 bool WidgetFactory::FreeWidgetGroup(std::string group_name) {
170  WIDGET_GROUP_MAP::iterator it = widget_group_map.find(group_name);
171  if (it != widget_group_map.end()) {
172  WIDGET_FREED_MAP& group = it->second;
173  WIDGET_FREED_MAP::iterator freed_it = group.begin();
174  while (freed_it != group.end()) {
175  int eid = freed_it->first;
176  if (GetValidById(eid) && !freed_it->second) {
177  delete GetObjectById(eid);
178  }
179  freed_it->second = true;
180  ++freed_it;
181  }
182  return true;
183  }
184  return false;
185 }
