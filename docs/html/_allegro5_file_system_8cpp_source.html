19 FSMode GetFSModeFromAllegroFileMode(uint32_t amode) {
20  int flags = 0;
21  if (amode & ALLEGRO_FILEMODE_READ) {flags |= FS_MODE_READ;}
22  if (amode & ALLEGRO_FILEMODE_WRITE) {flags |= FS_MODE_WRITE;}
23  if (amode & ALLEGRO_FILEMODE_EXECUTE) {flags |= FS_MODE_EXECUTE;}
24  if (amode & ALLEGRO_FILEMODE_HIDDEN) {flags |= FS_MODE_HIDDEN;}
25  if (!(amode & ALLEGRO_FILEMODE_ISDIR)) {flags |= FS_MODE_FILE;}
26  return FSMode(flags);
27 }

66 void Allegro5FileSystem::ReadDirectoryContents(Folder* folder , bool descend) {
67  if (!folder) {
68  throw EagleException("Allegro5FileSystem::ReadDirectory - folder is NULL!\n");
69  }
70  ArchiveFile* archive = dynamic_cast<ArchiveFile*>(folder);
71  ALLEGRO_FS_ENTRY* fs = 0;
72 
73  std::string p = folder->Path();
74  if (archive) {
75  if (!archive_mounted) {
76  MountArchive(p);
77  p = "";/// This is now root
78  }
79  else {
80  /// There is already an archive mounted here
81  EagleWarn() << "Allegro5FileSystem::ReadDirectoryContents - reading inside nested archives is not currently possible" << std::endl;
82  return;
83  }
84  }
85  else {
86  if (!(folder->Info().Mode().IsDir())) {
87  throw EagleException(StringPrintF(
88  "Allegro5FileSystem::ReadDirectoryContents - folder entry '%s' is not a directory!\n" , folder->Path().c_str()));
89  }
90  }
91 
92  fs = al_create_fs_entry(p.c_str());
93 
94  if (!fs) {
95  throw EagleException(StringPrintF(
96  "Allegro5FileSystem::ReadDirectoryContents - failed to read %s path '%s'\n" , archive?"archive":"folder" , p.c_str()));
97  }
98 
99  ResourceLibrary* reslib = Eagle::EagleLibrary::System("Any")->GetResourceLibrary();
100 
101  if (al_open_directory(fs)) {
102  ALLEGRO_FS_ENTRY* f = 0;
103  while ((f = al_read_directory(fs))) {
104  FSInfo info = GetFSInfo(f);
105  if (info.Mode().IsFile()) {
106  std::string ext = GetFileExt(GetFileName(info.Path()));
107  if (reslib->DeduceResourceType(ext) == RT_ARCHIVE) {
108  RegisterArchiveFile(folder , std::shared_ptr<ArchiveFile>(new ArchiveFile(info)));
109  }
110  else {
111  RegisterFile(folder , std::shared_ptr<File>(new File(info)));
112  }
113  }
114  else if (info.Mode().IsDir()) {
115  RegisterSubFolder(folder , std::shared_ptr<Folder>(new Folder(info)));
116  }
117  al_destroy_fs_entry(f);
118  }
119  }
120 
121  al_destroy_fs_entry(fs);
122 
123  if (descend) {
124  Folder::SUBFOLDERMAP subfoldermap = folder->SubFolders();
125  Folder::SUBFOLDERMAP::iterator it = subfoldermap.begin();
126  while (it != subfoldermap.end()) {
127  Folder* sf = it->second.get();
128  ReadDirectoryContents(sf , true);
129  ++it;
130  }
131  Folder::ARCHIVEMAP archivemap = folder->Archives();
132  Folder::ARCHIVEMAP::iterator ait = archivemap.begin();
133  while (ait != archivemap.end()) {
134  ArchiveFile* af = ait->second.get();
135  ReadDirectoryContents(af , true);
136  ++ait;
137  }
138  }
139 
140  if (archive) {
141  UnmountArchive();
142  }
143 }
