127 EagleImage* Allegro5Image::Clone(EagleGraphicsContext* parent_window , std::string iname) {
128  EagleImage* newimg = parent_window->EmptyImage(iname);
129  parent_window->DrawToBackBuffer();/// This sets the parent_window as the owning context for the call to Allocate
130  if (newimg->Allocate(W() , H() , ImageType())) {
131  parent_window->PushDrawingTarget(newimg);
132  parent_window->SetCopyBlender();
133  parent_window->Draw(this , 0 , 0);
134  parent_window->RestoreLastBlendingState();
135  parent_window->PopDrawingTarget();
136  return newimg;
137  }
138  /// else, fail
139  parent_window->FreeImage(newimg);
140 
141  return (EagleImage*)0;
142 }

162 bool Allegro5Image::Allocate(int width , int height , IMAGE_TYPE type) {
163  Free();
164  int area = width*height;

166 
167  if (type == SYSTEM_IMAGE) {type = VIDEO_IMAGE;}

189 bool Allegro5Image::Load(std::string file , IMAGE_TYPE type) {
190  Free();
191  if (type == SYSTEM_IMAGE) {type = VIDEO_IMAGE;}
192 
193  if (type == VIDEO_IMAGE) {
194  al_set_new_bitmap_flags(ALLEGRO_VIDEO_BITMAP);
195  }
196  else if (type == MEMORY_IMAGE) {
197  al_set_new_bitmap_flags(ALLEGRO_MEMORY_BITMAP);
198  }
199  bmp = al_load_bitmap(file.c_str());
200  if (!bmp) {
201  EagleError() << "Failed to load " << file << " from disk." << std::endl;
202  EAGLE_ASSERT(0);
203  }
204  else {
205  source = file;
206  pcontext = GetAllegro5WindowManager()->GetAssociatedContext(al_get_current_display());
207 /// SetName(file);

209  h = al_get_bitmap_height(bmp);
210  EagleInfo() << StringPrintF("Loaded image %s from file at size %d x %d." , file.c_str() , w , h) << std::endl;
211  }
212  return bmp;
