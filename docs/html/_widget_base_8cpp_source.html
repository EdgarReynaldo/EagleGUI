127 void WidgetBase::OnSelfFlagChanged(WidgetFlags new_widget_flags) {
128  unsigned int diff = FlagDiff(wflags , new_widget_flags);
129  if (diff & VISIBLE) {
130  new_widget_flags.AddFlag(NEEDS_BG_REDRAW);
131  }
132  if (diff & (HOVER | HASFOCUS)) {
133  new_widget_flags.AddFlag(NEEDS_BG_REDRAW);
134  }
135  if (new_widget_flags.FlagOn(NEEDS_BG_REDRAW)) {
136  new_widget_flags.AddFlag(NEEDS_REDRAW);
137  }
138  wflags.SetNewFlags(new_widget_flags);
139 
140  unsigned int cflags = wflags.ChangedFlags();
141 
142  if (!cflags) {return;}/// Nothing changed
143 
144  for (unsigned int i = 0 ; i < NUM_WIDGET_FLAGS - 1 ; ++i) {
145  /// If a flag changes, call its setter
146  unsigned int flag = 1 << i;
147  bool change = cflags & flag;
148  if (change) {
149  OnFlagChanged((WIDGET_FLAGS)flag , wflags.FlagOn((WIDGET_FLAGS)flag));
150  }
151  }
152 
153  if (cflags & NEEDS_REDRAW) {
154  SetRedrawFlag();
155  }
156  if ((cflags & NEEDS_BG_REDRAW) || (cflags & VISIBLE)) {
157  if (whandler) {
158  whandler->MakeAreaDirty(OuterArea());
159  }
160  }
161 
162  if (!wparent) {return;}
163 
164  if (cflags & HASFOCUS) {
165  wparent->SetFocusState(wflags.FlagOn(HASFOCUS));
166  }
167  if (cflags & HOVER) {
168  wparent->SetHoverState(wflags.FlagOn(HOVER));
169  }
170 }
