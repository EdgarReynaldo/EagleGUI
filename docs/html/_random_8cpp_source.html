68 double MTRNG::DRand() {
69  ///< A double is 8 bytes of precision, and our rng outputs 4 bytes at a time, combine two random numbers to make a double
70  MTYPE hiword = mt();
71  MTYPE loword = mt();
72  double d = ((unsigned long long)hiword << 32) | (unsigned long long)loword;
73  if (std::isinf(d) || std::isnan(d)) {
74  return DRand();///< Try again
75  }
76  return d;
77 }
